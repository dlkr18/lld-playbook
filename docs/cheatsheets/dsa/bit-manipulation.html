03 — Single Number (LC 136)

Every element appears twice except one. Find the unique element.

Key Insight: XOR all elements. Pairs cancel (a^a=0), leaving the single number.
#include <vector>
using namespace std;

class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int result = 0;
        for (int n : nums)
            result ^= n;
        return result;
    }
};

Time: O(n) • Space: O(1)

04 — Single Number II (LC 137)

Every element appears three times except one. Find the unique element.

Key Insight: Count bits at each position mod 3. The remainder is the single number's bit.
#include <vector>
using namespace std;

class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int result = 0;
        for (int i = 0; i < 32; i++) {
            int bitSum = 0;
            for (int n : nums)
                bitSum += (n >> i) & 1;
            if (bitSum % 3 != 0)
                result |= (1 << i);
        }
        return result;
    }
};

Time: O(32n) = O(n) • Space: O(1)

Alternative: State Machine (ones/twos)

class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int ones = 0, twos = 0;
        for (int n : nums) {
            ones = (ones ^ n) & ~twos;
            twos = (twos ^ n) & ~ones;
        }
        return ones;
    }
};

ones holds bits that appeared 1 mod 3 times, twos holds 2 mod 3.

05 — Number of 1 Bits (LC 191)

Count the number of set bits (Hamming weight) of an unsigned integer.

Brian Kernighan's Trick: n & (n-1) clears the lowest set bit. Loop until n == 0.
#include <cstdint>

class Solution {
public:
    int hammingWeight(uint32_t n) {
        int count = 0;
        while (n) {
            n &= (n - 1);  // clear lowest set bit
            count++;
        }
        return count;
    }
};

Time: O(number of set bits) • Space: O(1)

06 — Counting Bits (LC 338)

Return an array where ans[i] = number of 1 bits in i, for 0 ≤ i ≤ n.

DP Insight: dp[i] = dp[i & (i-1)] + 1. Removing one set bit gives a smaller solved subproblem.
#include <vector>
using namespace std;

class Solution {
public:
    vector<int> countBits(int n) {
        vector<int> dp(n + 1, 0);
        for (int i = 1; i <= n; i++)
            dp[i] = dp[i & (i - 1)] + 1;
        return dp;
    }
};

Time: O(n) • Space: O(n)

07 — Reverse Bits (LC 190)

Reverse all 32 bits of an unsigned integer.

Approach: Extract each bit from the right, place it from the left into the result.
#include <cstdint>

class Solution {
public:
    uint32_t reverseBits(uint32_t n) {
        uint32_t result = 0;
        for (int i = 0; i < 32; i++) {
            result = (result << 1) | (n & 1);
            n >>= 1;
        }
        return result;
    }
};

Time: O(32) = O(1) • Space: O(1)

08 — Missing Number (LC 268)

Array of n numbers from [0, n] with one missing. Find it.

XOR Trick: XOR all array elements with all indices [0..n]. Pairs cancel, missing number remains.
#include <vector>
using namespace std;

class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int n = nums.size();
        int xorAll = n;  // start with n (index n is not in array)
        for (int i = 0; i < n; i++)
            xorAll ^= i ^ nums[i];
        return xorAll;
    }
};

Time: O(n) • Space: O(1). No overflow risk unlike sum approach.

09 — Bitmask Subsets

Iterate all subsets of a bitmask. Essential for bitmask DP.

All subsets of a set of n elements

#include <vector>
#include <iostream>
using namespace std;

void allSubsets(int n) {
    // enumerate all 2^n subsets
    for (int mask = 0; mask < (1 << n); mask++) {
        // mask represents a subset; bit i set = element i included
        for (int i = 0; i < n; i++)
            if (mask & (1 << i))
                cout << i << " ";
        cout << "\n";
    }
}

All submasks of a given mask

void allSubmasks(int mask) {
    // iterates all submasks of mask in decreasing order
    for (int sub = mask; sub > 0; sub = (sub - 1) & mask) {
        // process submask 'sub'
    }
    // don't forget sub = 0 (empty set) if needed
}
Complexity: Iterating all submasks of all masks is O(3n) total, not O(4n). Each element is in/out/irrelevant.

Bitmask DP Template (TSP-style)

#include <vector>
#include <climits>
using namespace std;

int tsp(vector<vector<int>>& dist) {
    int n = dist.size();
    vector<vector<int>> dp(1 << n, vector<int>(n, INT_MAX));
    dp[1][0] = 0;  // start at node 0

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u)) || dp[mask][u] == INT_MAX) continue;
            for (int v = 0; v < n; v++) {
                if (mask & (1 << v)) continue;
                int next = mask | (1 << v);
                dp[next][v] = min(dp[next][v], dp[mask][u] + dist[u][v]);
            }
        }
    }

    int full = (1 << n) - 1;
    int ans = INT_MAX;
    for (int u = 1; u < n; u++)
        if (dp[full][u] != INT_MAX)
            ans = min(ans, dp[full][u] + dist[u][0]);
    return ans;
}
