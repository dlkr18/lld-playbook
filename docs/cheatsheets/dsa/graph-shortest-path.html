02 — Dijkstra's Algorithm

Core Idea

Greedy BFS: always expand the nearest unfinished node. Once a node is popped from the min-heap, its shortest distance is finalized.

  • Constraint: No negative edge weights
  • Data structure: Min-heap (priority_queue)

Template (Adjacency List + Min-Heap)

vector<int> dijkstra(int n, vector<vector<pair<int,int>>>& adj, int src) {
    vector<int> dist(n, INT_MAX);
    priority_queue<pair<int,int>,
                   vector<pair<int,int>>,
                   greater<>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue; // stale entry

        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

Network Delay Time (LC 743)

int networkDelayTime(vector<vector<int>>& times, int n, int k) {
    vector<vector<pair<int,int>>> adj(n + 1);
    for (auto& t : times)
        adj[t[0]].push_back({t[1], t[2]});

    auto dist = dijkstra(n + 1, adj, k);

    int ans = 0;
    for (int i = 1; i <= n; i++) {
        if (dist[i] == INT_MAX) return -1;
        ans = max(ans, dist[i]);
    }
    return ans;
}

Common Pitfalls

  • Stale check: Always skip if (d > dist[u]) after pop — avoids processing outdated entries
  • Negative edges: Dijkstra gives WRONG answers with negative weights
  • 1-indexed: Many LC problems use 1-indexed nodes — size your arrays as n + 1

03 — Bellman-Ford

Core Idea

Relax all edges V-1 times. After V-1 rounds, all shortest paths are found (if no negative cycle). One more round to detect negative cycles.

Template

vector<int> bellmanFord(int n, vector<vector<int>>& edges, int src) {
    vector<int> dist(n, INT_MAX);
    dist[src] = 0;

    for (int i = 0; i < n - 1; i++) {
        for (auto& e : edges) {
            int u = e[0], v = e[1], w = e[2];
            if (dist[u] != INT_MAX && dist[u] + w < dist[v])
                dist[v] = dist[u] + w;
        }
    }

    // Negative cycle detection (optional)
    for (auto& e : edges) {
        if (dist[e[0]] != INT_MAX && dist[e[0]] + e[2] < dist[e[1]])
            return {}; // negative cycle exists
    }
    return dist;
}

Key Points

PropertyDetail
Handles negative weights?Yes (unlike Dijkstra)
Detects negative cycles?Yes (extra round of relaxation)
TimeO(V · E) — slower than Dijkstra
When to use?Negative weights or K-stops constraint

04 — Floyd-Warshall

Core Idea

All-pairs shortest path. For each intermediate node k, check if going through k improves the path from i to j.

dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])

Template

void floydWarshall(vector<vector<int>>& dist, int n) {
    // dist[i][j] initialized: 0 if i==j, weight if edge, INF otherwise
    for (int k = 0; k < n; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (dist[i][k] != INT_MAX && dist[k][j] != INT_MAX)
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
}
Loop order matters: k must be the outermost loop. Think of it as “which intermediate nodes are allowed so far.”

Find the City (LC 1334)

Find the city with the smallest number of reachable cities within a distance threshold.

int findTheCity(int n, vector<vector<int>>& edges, int threshold) {
    vector<vector<int>> dist(n, vector<int>(n, INT_MAX/2));
    for (int i = 0; i < n; i++) dist[i][i] = 0;
    for (auto& e : edges) {
        dist[e[0]][e[1]] = e[2];
        dist[e[1]][e[0]] = e[2];
    }

    floydWarshall(dist, n);

    int ans = -1, minReach = n + 1;
    for (int i = 0; i < n; i++) {
        int cnt = 0;
        for (int j = 0; j < n; j++)
            if (i != j && dist[i][j] <= threshold) cnt++;
        if (cnt <= minReach) { minReach = cnt; ans = i; }
    }
    return ans;
}

05 — 0-1 BFS

Core Idea

When edge weights are only 0 or 1, use a deque instead of a priority queue. Push 0-weight edges to front, 1-weight edges to back.

Time: O(V + E) — faster than Dijkstra's O((V+E) log V).

Template

vector<int> bfs01(int n, vector<vector<pair<int,int>>>& adj, int src) {
    vector<int> dist(n, INT_MAX);
    deque<int> dq;
    dist[src] = 0;
    dq.push_back(src);

    while (!dq.empty()) {
        int u = dq.front(); dq.pop_front();
        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                if (w == 0) dq.push_front(v);
                else       dq.push_back(v);
            }
        }
    }
    return dist;
}

Minimum Obstacle Removal (LC 2290)

Grid with 0 (empty) and 1 (obstacle). Cost to enter an obstacle cell = 1. Classic 0-1 BFS.

int minimumObstacles(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    vector<vector<int>> dist(m, vector<int>(n, INT_MAX));
    deque<pair<int,int>> dq;
    dist[0][0] = 0;
    dq.push_back({0, 0});

    int dx[] = {0,1,0,-1}, dy[] = {1,0,-1,0};
    while (!dq.empty()) {
        auto [r, c] = dq.front(); dq.pop_front();
        for (int d = 0; d < 4; d++) {
            int x = r + dx[d], y = c + dy[d];
            if (x < 0 || x >= m || y < 0 || y >= n) continue;
            int w = grid[x][y];
            if (dist[r][c] + w < dist[x][y]) {
                dist[x][y] = dist[r][c] + w;
                if (w == 0) dq.push_front({x, y});
                else       dq.push_back({x, y});
            }
        }
    }
    return dist[m-1][n-1];
}

06 — Swim in Rising Water (LC 778)

Core Idea

Minimize the max elevation along any path from (0,0) to (n-1,n-1). Modified Dijkstra: instead of summing weights, take max.

Modified Dijkstra

int swimInWater(vector<vector<int>>& grid) {
    int n = grid.size();
    vector<vector<int>> dist(n, vector<int>(n, INT_MAX));
    priority_queue<tuple<int,int,int>,
                   vector<tuple<int,int,int>>,
                   greater<>> pq;

    dist[0][0] = grid[0][0];
    pq.push({grid[0][0], 0, 0});

    int dx[] = {0,1,0,-1}, dy[] = {1,0,-1,0};
    while (!pq.empty()) {
        auto [cost, r, c] = pq.top(); pq.pop();
        if (r == n-1 && c == n-1) return cost;
        if (cost > dist[r][c]) continue;

        for (int d = 0; d < 4; d++) {
            int x = r + dx[d], y = c + dy[d];
            if (x < 0 || x >= n || y < 0 || y >= n) continue;
            int newCost = max(cost, grid[x][y]);
            if (newCost < dist[x][y]) {
                dist[x][y] = newCost;
                pq.push({newCost, x, y});
            }
        }
    }
    return -1;
}
Key line: newCost = max(cost, grid[x][y]) — not cost + grid[x][y]. We minimize the bottleneck, not the sum.

07 — Cheapest Flights in K Stops (LC 787)

Core Idea

Bellman-Ford with exactly K+1 rounds (K stops = K+1 edges). Use a copy of dist to avoid chaining relaxations within a single round.

int findCheapestPrice(int n, vector<vector<int>>& flights,
                     int src, int dst, int k) {
    vector<int> dist(n, INT_MAX);
    dist[src] = 0;

    for (int i = 0; i <= k; i++) {
        vector<int> temp(dist); // snapshot before this round
        for (auto& f : flights) {
            int u = f[0], v = f[1], w = f[2];
            if (dist[u] != INT_MAX && dist[u] + w < temp[v])
                temp[v] = dist[u] + w;
        }
        dist = temp;
    }
    return dist[dst] == INT_MAX ? -1 : dist[dst];
}
Why copy? Without the snapshot, one round might chain multiple edges (violating the K-stops constraint).
