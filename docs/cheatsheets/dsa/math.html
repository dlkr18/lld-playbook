02. Sieve of Eratosthenes — Count Primes (LC 204)

Problem

Count the number of primes strictly less than n.

Approach

  • Create boolean array of size n, mark 0 and 1 as non-prime
  • For each i from 2 to √n: if prime, mark all multiples starting from i² as composite
  • Count remaining true entries
// LC 204: Count Primes — Sieve of Eratosthenes O(n log log n)
#include <vector>
using namespace std;

class Solution {
public:
    int countPrimes(int n) {
        if (n <= 2) return 0;
        vector<bool> is_prime(n, true);
        is_prime[0] = is_prime[1] = false;

        for (int i = 2; (long long)i * i < n; ++i) {
            if (is_prime[i]) {
                for (int j = i * i; j < n; j += i)
                    is_prime[j] = false;
            }
        }

        int count = 0;
        for (int i = 2; i < n; ++i)
            if (is_prime[i]) ++count;
        return count;
    }
};
Why start at i²? All composites less than i² would have already been marked by a smaller prime factor.

03. GCD & LCM — Euclid's Algorithm

Core Template

// Euclidean GCD — O(log(min(a,b)))
int gcd(int a, int b) {
    while (b) {
        a %= b;
        swap(a, b);
    }
    return a;
}

// LCM using GCD: lcm(a,b) = a / gcd(a,b) * b  (divide first to avoid overflow)
long long lcm(long long a, long long b) {
    return a / gcd(a, b) * b;
}

// Extended GCD: finds x, y such that a*x + b*y = gcd(a,b)
int extgcd(int a, int b, int& x, int& y) {
    if (b == 0) { x = 1; y = 0; return a; }
    int x1, y1;
    int g = extgcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - (a / b) * y1;
    return g;
}

Applications

  • Array GCD: gcd(gcd(a,b), c) — accumulate pairwise
  • Reduce fraction: divide both num/den by gcd
  • Modular inverse: use extended GCD when m is not prime
  • Water jug problem: solvable iff target % gcd(a,b) == 0

GCD of Array (LC 1979 variant)

#include <vector>
#include <numeric>
using namespace std;

int findGCD(vector<int>& nums) {
    int result = nums[0];
    for (int i = 1; i < nums.size(); ++i)
        result = __gcd(result, nums[i]);
    return result;
}

04. Fast Exponentiation — Pow(x, n) (LC 50)

Binary Exponentiation

Idea: x^n = (x^(n/2))² if n even, x · x^(n-1) if n odd. Reduces O(n) → O(log n).

// LC 50: Pow(x, n) — O(log n)
#include <cmath>
using namespace std;

class Solution {
public:
    double myPow(double x, int n) {
        long long N = n; // handle INT_MIN overflow
        if (N < 0) { x = 1.0 / x; N = -N; }

        double result = 1.0;
        while (N > 0) {
            if (N & 1) result *= x;
            x *= x;
            N >>= 1;
        }
        return result;
    }
};

Modular Exponentiation

// Compute (base^exp) % mod — avoids overflow with long long
long long power_mod(long long base, long long exp, long long mod) {
    long long result = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) result = result * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return result;
}
Key edge case: When n = INT_MIN, -n overflows int. Always cast to long long first.

05. Modular Arithmetic

Core Properties

IdentityFormula
Addition(a + b) % m = ((a % m) + (b % m)) % m
Subtraction(a - b) % m = ((a % m) - (b % m) + m) % m
Multiplication(a * b) % m = ((a % m) * (b % m)) % m
Division(a / b) % m = (a * b⁻¹) % m where b⁻¹ is modular inverse

Fermat's Little Theorem

If p is prime and gcd(a, p) = 1, then a^(p-1) ≡ 1 (mod p).
Therefore: a⁻¹ ≡ a^(p-2) (mod p)

// Modular inverse using Fermat's little theorem (p must be prime)
long long mod_inverse(long long a, long long p) {
    return power_mod(a, p - 2, p);
}

// Modular division: (a / b) % p = a * mod_inverse(b, p) % p
long long mod_divide(long long a, long long b, long long p) {
    return a % p * mod_inverse(b, p) % p;
}

Common MOD value

const int MOD = 1e9 + 7; // 10^9 + 7 (prime, fits in int)

// Safe modular addition
long long mod_add(long long a, long long b) {
    return ((a % MOD) + (b % MOD)) % MOD;
}

// Safe modular multiplication
long long mod_mul(long long a, long long b) {
    return (a % MOD) * (b % MOD) % MOD;
}

// Safe modular subtraction (avoids negative)
long long mod_sub(long long a, long long b) {
    return ((a % MOD) - (b % MOD) + MOD) % MOD;
}
When to use MOD: Any problem saying "return answer modulo 10^9+7" — apply mod at every intermediate multiplication to prevent overflow.

06. Prime Factorization — O(√n)

Template

#include <vector>
using namespace std;

vector<pair<int,int>> prime_factors(int n) {
    vector<pair<int,int>> factors; // {prime, exponent}

    for (int d = 2; (long long)d * d <= n; ++d) {
        if (n % d == 0) {
            int cnt = 0;
            while (n % d == 0) {
                n /= d;
                ++cnt;
            }
            factors.push_back({d, cnt});
        }
    }
    if (n > 1) factors.push_back({n, 1}); // remaining prime
    return factors;
}

// Count total divisors: product of (exp+1) for each prime factor
int count_divisors(int n) {
    int count = 1;
    for (int d = 2; (long long)d * d <= n; ++d) {
        int exp = 0;
        while (n % d == 0) { n /= d; ++exp; }
        count *= (exp + 1);
    }
    if (n > 1) count *= 2;
    return count;
}

Optimized: Smallest Prime Factor (SPF) via Sieve

// Precompute SPF for O(log n) factorization per query
vector<int> spf(1000001);

void build_spf() {
    for (int i = 0; i <= 1000000; ++i) spf[i] = i;
    for (int i = 2; (long long)i * i <= 1000000; ++i) {
        if (spf[i] == i) { // i is prime
            for (int j = i * i; j <= 1000000; j += i)
                if (spf[j] == j) spf[j] = i;
        }
    }
}

vector<pair<int,int>> factorize_fast(int n) {
    vector<pair<int,int>> factors;
    while (n > 1) {
        int p = spf[n], cnt = 0;
        while (n % p == 0) { n /= p; ++cnt; }
        factors.push_back({p, cnt});
    }
    return factors;
}

07. Pascal's Triangle (LC 118) + nCr

LC 118: Generate Pascal's Triangle

#include <vector>
using namespace std;

class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> tri(numRows);
        for (int i = 0; i < numRows; ++i) {
            tri[i].assign(i + 1, 1);
            for (int j = 1; j < i; ++j)
                tri[i][j] = tri[i-1][j-1] + tri[i-1][j];
        }
        return tri;
    }
};

nCr Computation (modular)

const int MOD = 1e9 + 7;
const int MAXN = 200001;
long long fact[MAXN], inv_fact[MAXN];

long long power_mod(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) res = res * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return res;
}

void precompute() {
    fact[0] = 1;
    for (int i = 1; i < MAXN; ++i)
        fact[i] = fact[i-1] * i % MOD;
    inv_fact[MAXN-1] = power_mod(fact[MAXN-1], MOD - 2, MOD);
    for (int i = MAXN-2; i >= 0; --i)
        inv_fact[i] = inv_fact[i+1] * (i+1) % MOD;
}

long long nCr(int n, int r) {
    if (r < 0 || r > n) return 0;
    return fact[n] % MOD * inv_fact[r] % MOD * inv_fact[n-r] % MOD;
}
Trick: Precompute factorials + inverse factorials once → O(1) per nCr query. Inverse factorials computed backwards using inv_fact[i] = inv_fact[i+1] * (i+1).

08. Count Trailing Zeros in Factorial (LC 172)

Insight

Trailing zeros come from factors of 10 = 2 × 5. Since factors of 2 are always more abundant, count factors of 5 in n!.

// LC 172: Factorial Trailing Zeroes — O(log₅ n)
class Solution {
public:
    int trailingZeroes(int n) {
        int count = 0;
        while (n >= 5) {
            n /= 5;
            count += n;
        }
        return count;
    }
};

Why divide repeatedly?

  • n/5 counts multiples of 5 (each contributes one 5)
  • n/25 counts multiples of 25 (contribute an extra 5)
  • n/125 counts multiples of 125, and so on...

Example: 100! → 100/5 + 100/25 + 100/125 = 20 + 4 + 0 = 24 trailing zeros

09. Happy Number (LC 202)

Problem

A number is "happy" if repeatedly replacing it with the sum of squares of its digits eventually reaches 1. Detect cycles using Floyd's cycle detection.

// LC 202: Happy Number — Floyd's cycle detection on digit squares
class Solution {
public:
    int digitSquareSum(int n) {
        int sum = 0;
        while (n > 0) {
            int d = n % 10;
            sum += d * d;
            n /= 10;
        }
        return sum;
    }

    bool isHappy(int n) {
        int slow = n, fast = n;
        do {
            slow = digitSquareSum(slow);
            fast = digitSquareSum(digitSquareSum(fast));
        } while (slow != fast);
        return slow == 1;
    }
};
Why Floyd's works here: The sequence is deterministic and bounded (for any int, digit-square-sum ≤ 81*10 = 810), so it must eventually cycle. If the cycle includes 1, it's happy; otherwise it's not.
