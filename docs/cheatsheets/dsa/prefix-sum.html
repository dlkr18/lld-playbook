Prefix Sum // & Difference Array

Precompute cumulative data so every range query is O(1). The difference array is the inverse — O(1) range updates, then reconstruct.

O(1) range query O(1) range update prefix + hashmap 2D extension

01 — Pattern Guide

Pick the right tool based on the operation mix:

ScenarioTechniqueWhy
Many range queries, array is static Prefix Sum O(n) build, O(1) per query
Count subarrays with sum = k Prefix + HashMap Turns subarray sum into two-sum lookup
Many range updates, one final read Difference Array O(1) per update, O(n) reconstruct
2D rectangle sum queries 2D Prefix Sum Inclusion-exclusion on 2D grid
Interleaved updates + queries BIT / Segment Tree Need both ops in O(log n)
Key insight: sum(l..r) = prefix[r+1] - prefix[l]. This identity drives everything.

02 — 1D Prefix Sum Template

Build

// prefix[i] = sum of nums[0..i-1]  (prefix[0] = 0)
vector<int> buildPrefix(vector<int>& nums) {
    int n = nums.size();
    vector<int> pre(n + 1, 0);
    for (int i = 0; i < n; i++)
        pre[i + 1] = pre[i] + nums[i];
    return pre;
}

Query — sum of nums[l..r] inclusive

int rangeSum(vector<int>& pre, int l, int r) {
    return pre[r + 1] - pre[l];
}
Convention: size n+1 prefix array avoids boundary checks. prefix[0] = 0 is the sentinel.

03 — Subarray Sum Equals K (LC 560)

Problem: Count the number of contiguous subarrays whose sum equals k.

Trick: sum(i..j) = prefix[j] - prefix[i-1] = k ⇒ we need prefix[i-1] = prefix[j] - k. Store running prefix counts in a hashmap.

int subarraySum(vector<int>& nums, int k) {
    unordered_map<int,int> freq;  // prefix_sum -> count
    freq[0] = 1;                    // empty prefix
    int sum = 0, ans = 0;
    for (int x : nums) {
        sum += x;
        if (freq.count(sum - k))
            ans += freq[sum - k];
        freq[sum]++;
    }
    return ans;
}

Time O(n), Space O(n). Works with negative numbers (sliding window can't).

04 — Range Sum Query Immutable (LC 303)

Problem: Given an immutable array, answer multiple sumRange(left, right) queries.

class NumArray {
    vector<int> pre;
public:
    NumArray(vector<int>& nums) {
        int n = nums.size();
        pre.resize(n + 1, 0);
        for (int i = 0; i < n; i++)
            pre[i + 1] = pre[i] + nums[i];
    }

    int sumRange(int left, int right) {
        return pre[right + 1] - pre[left];
    }
};

Build O(n), Query O(1). The textbook prefix sum application.

05 — Product of Array Except Self (LC 238)

Problem: Return array where ans[i] = product of all elements except nums[i]. No division allowed.

Trick: ans[i] = leftProduct[i] * rightProduct[i]. Build prefix product from left, suffix product from right.

vector<int> productExceptSelf(vector<int>& nums) {
    int n = nums.size();
    vector<int> ans(n, 1);

    // Left pass: ans[i] = product of nums[0..i-1]
    int left = 1;
    for (int i = 0; i < n; i++) {
        ans[i] = left;
        left *= nums[i];
    }

    // Right pass: multiply by product of nums[i+1..n-1]
    int right = 1;
    for (int i = n - 1; i >= 0; i--) {
        ans[i] *= right;
        right *= nums[i];
    }
    return ans;
}

Time O(n), Space O(1) extra (output array doesn't count). Two-pass prefix/suffix trick.

06 — 2D Prefix Sum Template

Build

Each cell pre[i][j] stores the sum of the rectangle from (0,0) to (i-1,j-1).

// matrix is m x n, pre is (m+1) x (n+1)
vector<vector<int>> build2D(vector<vector<int>>& mat) {
    int m = mat.size(), n = mat[0].size();
    vector<vector<int>> pre(m + 1, vector<int>(n + 1, 0));
    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++)
            pre[i][j] = mat[i-1][j-1]
                      + pre[i-1][j] + pre[i][j-1]
                      - pre[i-1][j-1];
    return pre;
}

Query — sum of rectangle (r1,c1) to (r2,c2) inclusive

int rectSum(vector<vector<int>>& pre,
            int r1, int c1, int r2, int c2) {
    return pre[r2+1][c2+1] - pre[r1][c2+1]
         - pre[r2+1][c1] + pre[r1][c1];
}
Inclusion-exclusion: Add the whole rectangle, subtract the two overlapping strips, add back the doubly-subtracted corner.

07 — Range Sum Query 2D (LC 304)

Problem: Given an immutable 2D matrix, answer multiple sumRegion(r1,c1,r2,c2) queries.

class NumMatrix {
    vector<vector<int>> pre;
public:
    NumMatrix(vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        pre.assign(m + 1, vector<int>(n + 1, 0));
        for (int i = 1; i <= m; i++)
            for (int j = 1; j <= n; j++)
                pre[i][j] = matrix[i-1][j-1]
                          + pre[i-1][j] + pre[i][j-1]
                          - pre[i-1][j-1];
    }

    int sumRegion(int r1, int c1, int r2, int c2) {
        return pre[r2+1][c2+1] - pre[r1][c2+1]
             - pre[r2+1][c1] + pre[r1][c1];
    }
};

Build O(m*n), Query O(1). Direct application of the 2D template above.

08 — Difference Array Template

Idea: To add val to every element in [l..r], just mark two endpoints. Reconstruct with a prefix sum at the end.

Range update in O(1)

vector<int> diff(n + 1, 0);  // size n+1 to avoid bounds check

// Add val to range [l..r] inclusive
void rangeAdd(vector<int>& diff, int l, int r, int val) {
    diff[l]   += val;
    diff[r+1] -= val;
}

Reconstruct final array

vector<int> reconstruct(vector<int>& diff, int n) {
    vector<int> arr(n);
    arr[0] = diff[0];
    for (int i = 1; i < n; i++)
        arr[i] = arr[i-1] + diff[i];
    return arr;
}
Duality: Prefix sum and difference array are inverses. prefixSum(diff) = original. diff(prefixSum) = original.

09 — Corporate Flight Bookings (LC 1109)

Problem: There are n flights (1-indexed). Each booking [first, last, seats] adds seats to flights first through last. Return total seats per flight.

vector<int> corpFlightBookings(
        vector<vector<int>>& bookings, int n) {
    vector<int> diff(n + 1, 0);

    for (auto& b : bookings) {
        int first = b[0], last = b[1], seats = b[2];
        diff[first - 1] += seats;    // convert 1-indexed to 0-indexed
        diff[last]      -= seats;    // r+1 in 0-indexed = last
    }

    // Reconstruct via prefix sum
    vector<int> ans(n);
    ans[0] = diff[0];
    for (int i = 1; i < n; i++)
        ans[i] = ans[i-1] + diff[i];
    return ans;
}

Time O(n + b) where b = number of bookings. Classic difference array — each booking is one O(1) update.
