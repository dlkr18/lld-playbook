01 — Pattern Guide

When to pick which paradigm:

SignalParadigmWhy
Sub-problems are independent, same structure, combine in O(n) D&C No overlapping work → recurse & merge
Sub-problems overlap, optimal substructure DP Cache to avoid re-computation
Locally optimal choice leads to global optimum Greedy No need to explore all sub-problems
Explore all choices, need to backtrack Backtracking No polynomial shortcut exists
D&C key insight: If you can divide into a sub-problems of size n/b and combine in O(nd), you have a D&C solution with complexity given by the Master Theorem.

D&C Template (3 Phases)

  1. Divide — split input into smaller sub-problems (usually halves)
  2. Conquer — solve each sub-problem recursively (base case stops recursion)
  3. Combine — merge sub-results into the final answer

03 — Merge Sort as D&C Template

Merge sort is the canonical D&C algorithm. Every merge-sort variant (count inversions, sort list, count smaller) follows this skeleton.

#include <vector>
using namespace std;

void merge(vector<int>& arr, int lo, int mid, int hi) {
    vector<int> tmp;
    int i = lo, j = mid + 1;
    while (i <= mid && j <= hi) {
        if (arr[i] <= arr[j]) tmp.push_back(arr[i++]);
        else                   tmp.push_back(arr[j++]);
    }
    while (i <= mid) tmp.push_back(arr[i++]);
    while (j <= hi)  tmp.push_back(arr[j++]);
    for (int k = lo; k <= hi; k++)
        arr[k] = tmp[k - lo];
}

void mergeSort(vector<int>& arr, int lo, int hi) {
    if (lo >= hi) return;          // base case
    int mid = lo + (hi - lo) / 2;
    mergeSort(arr, lo, mid);        // conquer left
    mergeSort(arr, mid + 1, hi);    // conquer right
    merge(arr, lo, mid, hi);        // combine
}

int main() {
    vector<int> arr = {5, 2, 4, 7, 1, 3, 6};
    mergeSort(arr, 0, arr.size() - 1);
    // arr is now {1, 2, 3, 4, 5, 6, 7}
    return 0;
}
Divide: split at mid. Conquer: recurse on each half. Combine: merge two sorted halves in O(n). T(n) = 2T(n/2) + O(n) = O(n log n).

04 — Pow(x, n) — LC 50

Binary Exponentiation. Compute xn in O(log n) instead of O(n).

Key idea: xn = (xn/2)2 if n is even, x · (xn/2)2 if n is odd.

#include <cmath>
#include <climits>

class Solution {
public:
    double myPow(double x, int n) {
        long long N = n;
        if (N < 0) { x = 1.0 / x; N = -N; }
        double ans = 1.0;
        while (N > 0) {
            if (N & 1) ans *= x;
            x *= x;
            N >>= 1;
        }
        return ans;
    }
};
AspectDetail
TimeO(log n)
SpaceO(1) iterative, O(log n) recursive
Edge casen = INT_MIN → use long long for negation
D&C linka=1, b=2, d=0 → Case 2 → O(log n)

05 — Maximum Subarray — LC 53

D&C Approach: O(n log n)

Max subarray is either entirely in the left half, entirely in the right half, or crosses the midpoint. Solve each and take the max.

#include <vector>
#include <algorithm>
#include <climits>
using namespace std;

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        return solve(nums, 0, nums.size() - 1);
    }

private:
    int solve(vector<int>& nums, int lo, int hi) {
        if (lo == hi) return nums[lo];
        int mid = lo + (hi - lo) / 2;
        int left  = solve(nums, lo, mid);
        int right = solve(nums, mid + 1, hi);
        int cross = crossMax(nums, lo, mid, hi);
        return max({left, right, cross});
    }

    int crossMax(vector<int>& nums, int lo, int mid, int hi) {
        int leftMax = INT_MIN, sum = 0;
        for (int i = mid; i >= lo; i--) {
            sum += nums[i];
            leftMax = max(leftMax, sum);
        }
        int rightMax = INT_MIN;
        sum = 0;
        for (int i = mid + 1; i <= hi; i++) {
            sum += nums[i];
            rightMax = max(rightMax, sum);
        }
        return leftMax + rightMax;
    }
};

Kadane's DP Approach: O(n) — preferred in interviews

int maxSubArray(vector<int>& nums) {
    int cur = nums[0], best = nums[0];
    for (int i = 1; i < (int)nums.size(); i++) {
        cur = max(nums[i], cur + nums[i]);
        best = max(best, cur);
    }
    return best;
}
ApproachTimeSpaceWhen

Kadane's DPO(n)O(1)Default — simpler & faster

06 — Sort List — LC 148

Merge sort on linked list. Unlike arrays, linked lists can be split in O(n) via slow/fast pointers and merged in O(1) space.

struct ListNode {
    int val;
    ListNode* next;
    ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (!head || !head->next) return head;

        // Divide: find mid with slow/fast
        ListNode* slow = head;
        ListNode* fast = head->next;
        while (fast && fast->next) {
            slow = slow->next;
            fast = fast->next->next;
        }
        ListNode* mid = slow->next;
        slow->next = nullptr;

        // Conquer
        ListNode* left  = sortList(head);
        ListNode* right = sortList(mid);

        // Combine
        return mergeLists(left, right);
    }

private:
    ListNode* mergeLists(ListNode* a, ListNode* b) {
        ListNode dummy(0);
        ListNode* tail = &dummy;
        while (a && b) {
            if (a->val <= b->val) { tail->next = a; a = a->next; }
            else                  { tail->next = b; b = b->next; }
            tail = tail->next;
        }
        tail->next = a ? a : b;
        return dummy.next;
    }
};
AspectDetail
TimeO(n log n)
SpaceO(log n) recursion stack (O(1) merge — no aux array needed)
Why D&C?Linked lists split/merge cheaply without random access

07 — Count of Smaller Numbers After Self — LC 315

Merge sort + count inversions. During the merge step, when a right element is placed before left elements, count how many left elements it “jumps over”.

#include <vector>
using namespace std;

class Solution {
    vector<int> count;

    void mergeSort(vector<pair<int,int>>& arr, int lo, int hi) {
        if (lo >= hi) return;
        int mid = lo + (hi - lo) / 2;
        mergeSort(arr, lo, mid);
        mergeSort(arr, mid + 1, hi);
        merge(arr, lo, mid, hi);
    }

    void merge(vector<pair<int,int>>& arr, int lo, int mid, int hi) {
        vector<pair<int,int>> tmp;
        int i = lo, j = mid + 1;
        int rightCount = 0;
        while (i <= mid && j <= hi) {
            if (arr[j].first < arr[i].first) {
                rightCount++;
                tmp.push_back(arr[j++]);
            } else {
                count[arr[i].second] += rightCount;
                tmp.push_back(arr[i++]);
            }
        }
        while (i <= mid) {
            count[arr[i].second] += rightCount;
            tmp.push_back(arr[i++]);
        }
        while (j <= hi) tmp.push_back(arr[j++]);
        for (int k = lo; k <= hi; k++)
            arr[k] = tmp[k - lo];
    }

public:
    vector<int> countSmaller(vector<int>& nums) {
        int n = nums.size();
        count.assign(n, 0);
        vector<pair<int,int>> arr(n);
        for (int i = 0; i < n; i++)
            arr[i] = {nums[i], i};
        mergeSort(arr, 0, n - 1);
        return count;
    }
};
Key trick: Pair each value with its original index. During merge, when a right element (smaller) is placed first, increment rightCount. When a left element is placed, it has rightCount smaller elements after it.
AspectDetail
TimeO(n log n)
SpaceO(n)
Also solvesLC 493 (Reverse Pairs), LC 327 (Count Range Sum)

08 — Median of Two Sorted Arrays — LC 4

Binary search D&C. Binary search on the partition position of the smaller array. Ensure the left partition of both arrays combined has exactly half the total elements.

#include <vector>
#include <algorithm>
#include <climits>
using namespace std;

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        if (nums1.size() > nums2.size())
            return findMedianSortedArrays(nums2, nums1);

        int m = nums1.size(), n = nums2.size();
        int lo = 0, hi = m;
        int half = (m + n + 1) / 2;

        while (lo <= hi) {
            int i = lo + (hi - lo) / 2;   // partition in nums1
            int j = half - i;               // partition in nums2

            int left1  = (i > 0) ? nums1[i - 1] : INT_MIN;
            int right1 = (i < m) ? nums1[i]     : INT_MAX;
            int left2  = (j > 0) ? nums2[j - 1] : INT_MIN;
            int right2 = (j < n) ? nums2[j]     : INT_MAX;

            if (left1 <= right2 && left2 <= right1) {
                if ((m + n) % 2 == 0)
                    return (max(left1, left2) + min(right1, right2)) / 2.0;
                return max(left1, left2);
            }
            if (left1 > right2) hi = i - 1;
            else                lo = i + 1;
        }
        return 0.0;
    }
};
AspectDetail
TimeO(log min(m, n))
SpaceO(1)
TrickAlways binary search on the smaller array
Invariantleft1 ≤ right2 and left2 ≤ right1

09 — Search a 2D Matrix II — LC 240

Staircase search. Start from the top-right (or bottom-left). Each comparison eliminates an entire row or column.

#include <vector>
using namespace std;

class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        if (matrix.empty()) return false;
        int rows = matrix.size(), cols = matrix[0].size();
        int r = 0, c = cols - 1;      // start top-right

        while (r < rows && c >= 0) {
            if (matrix[r][c] == target)
                return true;
            else if (matrix[r][c] > target)
                c--;                     // eliminate column
            else
                r++;                     // eliminate row
        }
        return false;
    }
};
AspectDetail
TimeO(m + n)
SpaceO(1)
D&C linkEach step divides the search space by one row or column
Also worksStarting from bottom-left (mirror logic)
Why top-right? Moving left decreases value, moving down increases it — giving a clear binary decision at each step.
