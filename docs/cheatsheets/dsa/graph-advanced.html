01 — Union-Find Template

DSU with Path Compression + Union by Rank

class DSU {
public:
    vector<int> parent, rank_;
    int components;

    DSU(int n) : parent(n), rank_(n, 0), components(n) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]); // path compression
        return parent[x];
    }

    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false; // already connected
        if (rank_[a] < rank_[b]) swap(a, b);
        parent[b] = a;
        if (rank_[a] == rank_[b]) rank_[a]++;
        components--;
        return true;
    }

    bool connected(int a, int b) {
        return find(a) == find(b);
    }
};

Complexity

OperationTimeWhy
find(x)O(α(n)) ≈ O(1)Path compression flattens tree
unite(a, b)O(α(n)) ≈ O(1)Union by rank keeps tree shallow
Build from n nodesO(n)Initialize parent array
α(n) is the inverse Ackermann function — effectively constant for all practical inputs (up to 1080).

02 — Union-Find Problems

Redundant Connection (LC 684)

Find the edge that, if removed, makes the graph a tree. The first edge that connects two already-connected nodes is the answer.

vector<int> findRedundantConnection(vector<vector<int>>& edges) {
    int n = edges.size();
    DSU dsu(n + 1);
    for (auto& e : edges) {
        if (!dsu.unite(e[0], e[1]))
            return e; // already connected = redundant
    }
    return {};
}

Number of Islands II (LC 305)

Stream of “add land” operations. After each, report number of islands. Union-Find gives O(1) per op.

vector<int> numIslands2(int m, int n, vector<vector<int>>& positions) {
    DSU dsu(m * n);
    dsu.components = 0;
    vector<vector<bool>> grid(m, vector<bool>(n, false));
    vector<int> result;
    int dx[] = {0,1,0,-1}, dy[] = {1,0,-1,0};

    for (auto& p : positions) {
        int r = p[0], c = p[1];
        if (grid[r][c]) { result.push_back(dsu.components); continue; }
        grid[r][c] = true;
        dsu.components++;
        for (int d = 0; d < 4; d++) {
            int x = r + dx[d], y = c + dy[d];
            if (x >= 0 && x < m && y >= 0 && y < n && grid[x][y])
                dsu.unite(r * n + c, x * n + y);
        }
        result.push_back(dsu.components);
    }
    return result;
}

Accounts Merge (LC 721)

Each account has emails. Merge accounts that share any email. Union-Find on email → parent mapping.

vector<vector<string>> accountsMerge(vector<vector<string>>& accounts) {
    unordered_map<string, int> emailToId;
    unordered_map<string, string> emailToName;
    int id = 0;

    for (auto& acc : accounts)
        for (int i = 1; i < acc.size(); i++) {
            if (!emailToId.count(acc[i]))
                emailToId[acc[i]] = id++;
            emailToName[acc[i]] = acc[0];
        }

    DSU dsu(id);
    for (auto& acc : accounts)
        for (int i = 2; i < acc.size(); i++)
            dsu.unite(emailToId[acc[1]], emailToId[acc[i]]);

    unordered_map<int, vector<string>> groups;
    for (auto& [email, eid] : emailToId)
        groups[dsu.find(eid)].push_back(email);

    vector<vector<string>> result;
    for (auto& [root, emails] : groups) {
        sort(emails.begin(), emails.end());
        emails.insert(emails.begin(), emailToName[emails[0]]);
        result.push_back(emails);
    }
    return result;
}

03 — Kruskal's MST

Core Idea

Sort all edges by weight. Greedily add the lightest edge that doesn't form a cycle (check with Union-Find). Stop after V-1 edges.

Template

int kruskalMST(int n, vector<vector<int>>& edges) {
    // edges[i] = {u, v, weight}
    sort(edges.begin(), edges.end(),
         [](auto& a, auto& b) { return a[2] < b[2]; });

    DSU dsu(n);
    int cost = 0, edgesUsed = 0;

    for (auto& e : edges) {
        if (dsu.unite(e[0], e[1])) {
            cost += e[2];
            if (++edgesUsed == n - 1) break;
        }
    }
    return edgesUsed == n - 1 ? cost : -1; // -1 if not connected
}

Min Cost to Connect All Points (LC 1584)

int minCostConnectPoints(vector<vector<int>>& points) {
    int n = points.size();
    vector<vector<int>> edges;
    for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++) {
            int d = abs(points[i][0] - points[j][0])
                  + abs(points[i][1] - points[j][1]);
            edges.push_back({i, j, d});
        }
    return kruskalMST(n, edges);
}

04 — Prim's MST

Core Idea

Start from any node. Greedily add the cheapest edge that connects the MST set to a new node. Uses a min-heap.

Template

int primMST(int n, vector<vector<pair<int,int>>>& adj) {
    vector<bool> inMST(n, false);
    priority_queue<pair<int,int>,
                   vector<pair<int,int>>,
                   greater<>> pq;

    pq.push({0, 0}); // {cost, node}
    int totalCost = 0, edgesUsed = 0;

    while (!pq.empty() && edgesUsed < n) {
        auto [cost, u] = pq.top(); pq.pop();
        if (inMST[u]) continue;

        inMST[u] = true;
        totalCost += cost;
        edgesUsed++;

        for (auto [v, w] : adj[u])
            if (!inMST[v])
                pq.push({w, v});
    }
    return totalCost;
}

06 — Multi-Source & Advanced Patterns

Graph Valid Tree (LC 261)

A graph is a valid tree if: (1) V-1 edges and (2) all nodes connected.

bool validTree(int n, vector<vector<int>>& edges) {
    if (edges.size() != n - 1) return false;
    DSU dsu(n);
    for (auto& e : edges)
        dsu.unite(e[0], e[1]);
    return dsu.components == 1;
}

Number of Connected Components (LC 323)

int countComponents(int n, vector<vector<int>>& edges) {
    DSU dsu(n);
    for (auto& e : edges)
        dsu.unite(e[0], e[1]);
    return dsu.components;
}

Earliest Moment Everyone Friends (LC 1101)

Sort logs by time, union friends. First time components == 1 is the answer.

int earliestAcq(vector<vector<int>>& logs, int n) {
    sort(logs.begin(), logs.end());
    DSU dsu(n);
    for (auto& l : logs) {
        dsu.unite(l[1], l[2]);
        if (dsu.components == 1) return l[0];
    }
    return -1;
}
