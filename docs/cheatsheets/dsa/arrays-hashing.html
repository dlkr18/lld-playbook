02 — Two Sum (LC 1)

Idea: For each number, check if its complement target - nums[i] already exists in a hashmap. One pass, O(n).

Key insight: Don't search for a pair — search for the complement.
vector<int> twoSum(vector<int>& nums, int target) {
    unordered_map<int, int> seen;  // value -> index
    for (int i = 0; i < nums.size(); i++) {
        int complement = target - nums[i];
        if (seen.count(complement)) {
            return {seen[complement], i};
        }
        seen[nums[i]] = i;
    }
    return {};
}
  • Time: O(n) — single pass
  • Space: O(n) — hashmap stores at most n entries
  • Edge case: duplicate values work because we check before inserting

03 — Group Anagrams (LC 49)

Idea: Two strings are anagrams if they have the same character frequency. Use a canonical key to group them.

Approach A: Sorted Key

vector<vector<string>> groupAnagrams(vector<string>& strs) {
    unordered_map<string, vector<string>> groups;
    for (auto& s : strs) {
        string key = s;
        sort(key.begin(), key.end());
        groups[key].push_back(s);
    }
    vector<vector<string>> result;
    for (auto& [k, v] : groups) {
        result.push_back(v);
    }
    return result;
}

Time: O(n · k log k) where k = max string length.

Approach B: Frequency Key (optimal for small alphabet)

vector<vector<string>> groupAnagrams(vector<string>& strs) {
    unordered_map<string, vector<string>> groups;
    for (auto& s : strs) {
        string key(26, '0');
        for (char c : s) key[c - 'a']++;
        groups[key].push_back(s);
    }
    vector<vector<string>> result;
    for (auto& [k, v] : groups) {
        result.push_back(v);
    }
    return result;
}

Time: O(n · k) — no sorting needed. Better when strings are long.

Which to pick? Frequency key is O(n·k) vs sorted O(n·k log k). Use frequency key for lowercase-only; sorted key for unicode.

04 — Top K Frequent Elements (LC 347)

Idea: Count frequencies, then use bucket sort — index = frequency, value = list of elements. Scan from right to collect top K.

Approach A: Bucket Sort — O(n)

vector<int> topKFrequent(vector<int>& nums, int k) {
    unordered_map<int, int> freq;
    for (int n : nums) freq[n]++;

    // buckets[i] = elements appearing exactly i times
    vector<vector<int>> buckets(nums.size() + 1);
    for (auto& [num, cnt] : freq) {
        buckets[cnt].push_back(num);
    }

    vector<int> result;
    for (int i = buckets.size() - 1; i >= 0 && result.size() < k; i--) {
        for (int num : buckets[i]) {
            result.push_back(num);
            if (result.size() == k) break;
        }
    }
    return result;
}

Approach B: Min-Heap — O(n log k)

vector<int> topKFrequent(vector<int>& nums, int k) {
    unordered_map<int, int> freq;
    for (int n : nums) freq[n]++;

    // min-heap: smallest freq on top, pop when size > k
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> minHeap;
    for (auto& [num, cnt] : freq) {
        minHeap.push({cnt, num});
        if (minHeap.size() > k) minHeap.pop();
    }

    vector<int> result;
    while (!minHeap.empty()) {
        result.push_back(minHeap.top().second);
        minHeap.pop();
    }
    return result;
}
Prefer bucket sort in interviews — it's O(n), simpler to explain, and shows you know the trick. Heap is the safe fallback.

05 — Longest Consecutive Sequence (LC 128)

Idea: Put everything in a HashSet. Only start counting from sequence heads — numbers where n-1 is NOT in the set. This ensures O(n) total.

Key insight: Without the head check, you'd re-traverse subsequences. The head check makes each element visited at most twice.
int longestConsecutive(vector<int>& nums) {
    unordered_set<int> numSet(nums.begin(), nums.end());
    int longest = 0;

    for (int n : numSet) {
        // only start from sequence heads
        if (!numSet.count(n - 1)) {
            int length = 1;
            while (numSet.count(n + length)) {
                length++;
            }
            longest = max(longest, length);
        }
    }
    return longest;
}
  • Time: O(n) — each element is visited at most twice (once in outer loop, once in while)
  • Space: O(n) — the HashSet
  • Why not sort? Sorting gives O(n log n). HashSet approach is O(n).

06 — Valid Anagram (LC 242)

Idea: Count character frequencies in s, decrement for t. If all counts are zero, they're anagrams.

bool isAnagram(string s, string t) {
    if (s.size() != t.size()) return false;

    int count[26] = {};
    for (int i = 0; i < s.size(); i++) {
        count[s[i] - 'a']++;
        count[t[i] - 'a']--;
    }
    for (int c : count) {
        if (c != 0) return false;
    }
    return true;
}
  • Time: O(n)
  • Space: O(1) — fixed 26-char array
  • Follow-up (Unicode): Use unordered_map<char, int> instead of fixed array

07 — Contains Duplicate (LC 217)

Idea: Insert into a HashSet; if insertion fails (already present), we have a duplicate.

bool containsDuplicate(vector<int>& nums) {
    unordered_set<int> seen;
    for (int n : nums) {
        if (!seen.insert(n).second) {
            return true;
        }
    }
    return false;
}
  • Time: O(n)
  • Space: O(n)
  • Alt: Sort first for O(n log n) time, O(1) space — mention as trade-off

08 — Product of Array Except Self (LC 238)

Idea: For each index, result = product of everything to its left × product of everything to its right. Two passes, no division.

Key insight: Build prefix products left-to-right, then multiply suffix products right-to-left. The output array itself stores prefixes.
vector<int> productExceptSelf(vector<int>& nums) {
    int n = nums.size();
    vector<int> result(n, 1);

    // Pass 1: result[i] = product of all elements to the left of i
    int prefix = 1;
    for (int i = 0; i < n; i++) {
        result[i] = prefix;
        prefix *= nums[i];
    }

    // Pass 2: multiply by product of all elements to the right of i
    int suffix = 1;
    for (int i = n - 1; i >= 0; i--) {
        result[i] *= suffix;
        suffix *= nums[i];
    }

    return result;
}
  • Time: O(n) — two passes
  • Space: O(1) extra — output array doesn't count per problem statement
  • Why no division? Division fails with zeros. The prefix/suffix approach handles zeros naturally.

Walkthrough: nums = [1, 2, 3, 4]

09 — Encode and Decode Strings (LC 271)

Idea: Use length-prefix encoding: len#string. This handles any character (including delimiters) inside strings.

Why not just join with a delimiter? The strings themselves might contain the delimiter. Length-prefix is unambiguous.
class Codec {
public:
    string encode(vector<string>& strs) {
        string encoded;
        for (auto& s : strs) {
            encoded += to_string(s.size()) + "#" + s;
        }
        return encoded;
    }

    vector<string> decode(string s) {
        vector<string> result;
        int i = 0;
        while (i < s.size()) {
            int hashPos = s.find('#', i);
            int len = stoi(s.substr(i, hashPos - i));
            result.push_back(s.substr(hashPos + 1, len));
            i = hashPos + 1 + len;
        }
        return result;
    }
};
  • Time: O(n) for both encode and decode, where n = total characters
  • Space: O(n)
  • Format example: ["hello","world"] → "5#hello5#world"

Walkthrough

InputEncodedDecode Steps
["ab","c#d",""] "2#ab3#c#d0#" read 2 → "ab", read 3 → "c#d", read 0 → ""
