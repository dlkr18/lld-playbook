Range queries + point/range updates in O(log n). BIT for simple prefix sums, Segment Tree for arbitrary associative operations + lazy propagation.

01 — Pattern Guide

When to use which data structure:

ScenarioUseWhy
Static array, range sum/min queries only Prefix Sum / Sparse Table O(1) query, no updates needed
Point update + prefix/range sum query BIT (Fenwick) Simpler, less memory, faster constant
Point update + range min/max/gcd query Segment Tree BIT can't handle non-invertible ops
Range update + range query Seg Tree + Lazy Lazy propagation for batch updates
Range update + point query BIT (difference array) BIT on difference array trick
2D range queries 2D BIT / 2D Seg Tree Extend 1D approach to grid
Rule of thumb: If BIT can solve it, use BIT. It's simpler, shorter, and has better constants. Reach for Segment Tree only when you need non-invertible operations or lazy propagation.

02 — BIT (Fenwick Tree) Template

Point update + prefix query. 1-indexed array. Core insight: i & (-i) gives the lowest set bit.

#include <vector>
using namespace std;

class BIT {
    vector<int> tree;
    int n;
public:
    BIT(int n) : n(n), tree(n + 1, 0) {}

    // Add val to index i (1-indexed)
    void update(int i, int val) {
        for (; i <= n; i += i & (-i))
            tree[i] += val;
    }

    // Prefix sum [1..i]
    int query(int i) {
        int sum = 0;
        for (; i > 0; i -= i & (-i))
            sum += tree[i];
        return sum;
    }

    // Range sum [l..r]
    int query(int l, int r) {
        return query(r) - query(l - 1);
    }
};
Memory: O(n). Key insight: parent of node i is i + (i & -i) for updates, i - (i & -i) for queries.

03 — Range Sum Query Mutable (LC 307)

Given a mutable integer array, handle: update(index, val) and sumRange(left, right).

#include <vector>
using namespace std;

class NumArray {
    vector<int> bit;
    vector<int> nums;
    int n;

    void add(int i, int delta) {
        for (++i; i <= n; i += i & (-i))
            bit[i] += delta;
    }

    int prefixSum(int i) {
        int s = 0;
        for (++i; i > 0; i -= i & (-i))
            s += bit[i];
        return s;
    }

public:
    NumArray(vector<int>& nums) : n(nums.size()), bit(nums.size() + 1, 0), nums(nums) {
        for (int i = 0; i < n; i++)
            add(i, nums[i]);
    }

    void update(int index, int val) {
        add(index, val - nums[index]);
        nums[index] = val;
    }

    int sumRange(int left, int right) {
        return prefixSum(right) - (left > 0 ? prefixSum(left - 1) : 0);
    }
};

Time: O(log n) per update/query. Space: O(n). Converts 0-indexed to 1-indexed internally.

04 — Segment Tree Template

Build, point update, range query. Iterative (bottom-up) for simplicity and speed.

Iterative Segment Tree (Sum)

#include <vector>
using namespace std;

class SegTree {
    int n;
    vector<long long> tree;
public:
    SegTree(vector<int>& arr) {
        n = arr.size();
        tree.assign(2 * n, 0);
        // Build: leaves at [n..2n-1]
        for (int i = 0; i < n; i++)
            tree[n + i] = arr[i];
        for (int i = n - 1; i >= 1; i--)
            tree[i] = tree[2 * i] + tree[2 * i + 1];
    }

    // Point update: set arr[pos] = val
    void update(int pos, int val) {
        pos += n;
        tree[pos] = val;
        for (pos /= 2; pos >= 1; pos /= 2)
            tree[pos] = tree[2 * pos] + tree[2 * pos + 1];
    }

    // Range query [l, r) — half-open
    long long query(int l, int r) {
        long long res = 0;
        for (l += n, r += n; l < r; l /= 2, r /= 2) {
            if (l & 1) res += tree[l++];
            if (r & 1) res += tree[--r];
        }
        return res;
    }
};

Recursive Segment Tree

#include <vector>
using namespace std;

class SegTreeRecursive {
    int n;
    vector<long long> tree;

    void build(vector<int>& arr, int node, int start, int end) {
        if (start == end) {
            tree[node] = arr[start];
            return;
        }
        int mid = (start + end) / 2;
        build(arr, 2 * node, start, mid);
        build(arr, 2 * node + 1, mid + 1, end);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    void update(int node, int start, int end, int idx, int val) {
        if (start == end) {
            tree[node] = val;
            return;
        }
        int mid = (start + end) / 2;
        if (idx <= mid)
            update(2 * node, start, mid, idx, val);
        else
            update(2 * node + 1, mid + 1, end, idx, val);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    long long query(int node, int start, int end, int l, int r) {
        if (r < start || end < l) return 0;
        if (l <= start && end <= r) return tree[node];
        int mid = (start + end) / 2;
        return query(2 * node, start, mid, l, r)
             + query(2 * node + 1, mid + 1, end, l, r);
    }

public:
    SegTreeRecursive(vector<int>& arr) {
        n = arr.size();
        tree.assign(4 * n, 0);
        build(arr, 1, 0, n - 1);
    }

    void update(int idx, int val) { update(1, 0, n - 1, idx, val); }
    long long query(int l, int r) { return query(1, 0, n - 1, l, r); }
};

05 — Lazy Propagation (Range Update + Range Query)

Range add update + range sum query. Lazy nodes defer updates to children until needed.

#include <vector>
using namespace std;

class LazySegTree {
    int n;
    vector<long long> tree, lazy;

    void push(int node, int start, int end) {
        if (lazy[node] != 0) {
            tree[node] += lazy[node] * (end - start + 1);
            if (start != end) {
                lazy[2 * node] += lazy[node];
                lazy[2 * node + 1] += lazy[node];
            }
            lazy[node] = 0;
        }
    }

    void build(vector<int>& arr, int node, int start, int end) {
        if (start == end) {
            tree[node] = arr[start];
            return;
        }
        int mid = (start + end) / 2;
        build(arr, 2 * node, start, mid);
        build(arr, 2 * node + 1, mid + 1, end);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    void rangeUpdate(int node, int start, int end, int l, int r, long long val) {
        push(node, start, end);
        if (r < start || end < l) return;
        if (l <= start && end <= r) {
            lazy[node] += val;
            push(node, start, end);
            return;
        }
        int mid = (start + end) / 2;
        rangeUpdate(2 * node, start, mid, l, r, val);
        rangeUpdate(2 * node + 1, mid + 1, end, l, r, val);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    long long rangeQuery(int node, int start, int end, int l, int r) {
        push(node, start, end);
        if (r < start || end < l) return 0;
        if (l <= start && end <= r) return tree[node];
        int mid = (start + end) / 2;
        return rangeQuery(2 * node, start, mid, l, r)
             + rangeQuery(2 * node + 1, mid + 1, end, l, r);
    }

public:
    LazySegTree(vector<int>& arr) {
        n = arr.size();
        tree.assign(4 * n, 0);
        lazy.assign(4 * n, 0);
        build(arr, 1, 0, n - 1);
    }

    void update(int l, int r, long long val) {
        rangeUpdate(1, 0, n - 1, l, r, val);
    }

    long long query(int l, int r) {
        return rangeQuery(1, 0, n - 1, l, r);
    }
};
Key idea: Don't propagate updates immediately. Store pending additions in lazy[] and push down only when a node is accessed.

06 — Range Minimum Query

Segment tree for minimum. Change merge from + to min. Identity element is INT_MAX.

#include <vector>
#include <climits>>
using namespace std;

class RMQ {
    int n;
    vector<int> tree;
public:
    RMQ(vector<int>& arr) {
        n = arr.size();
        tree.assign(2 * n, INT_MAX);
        for (int i = 0; i < n; i++)
            tree[n + i] = arr[i];
        for (int i = n - 1; i >= 1; i--)
            tree[i] = min(tree[2 * i], tree[2 * i + 1]);
    }

    void update(int pos, int val) {
        pos += n;
        tree[pos] = val;
        for (pos /= 2; pos >= 1; pos /= 2)
            tree[pos] = min(tree[2 * pos], tree[2 * pos + 1]);
    }

    // Query min in [l, r) — half-open
    int query(int l, int r) {
        int res = INT_MAX;
        for (l += n, r += n; l < r; l /= 2, r /= 2) {
            if (l & 1) res = min(res, tree[l++]);
            if (r & 1) res = min(res, tree[--r]);
        }
        return res;
    }
};

Same structure works for max (change min→max, identity to INT_MIN) or GCD (identity=0).

07 — Count of Range Sum (LC 327)

Count pairs (i,j) where lower ≤ sum(i..j) ≤ upper. Merge sort approach: count valid pairs during merge step on prefix sums.

#include <vector>
using namespace std;

class Solution {
    int count;
    int lower, upper;

    void mergeSort(vector<long long>& sums, int lo, int hi) {
        if (hi - lo <= 1) return;
        int mid = (lo + hi) / 2;
        mergeSort(sums, lo, mid);
        mergeSort(sums, mid, hi);

        // Count: for each i in [lo,mid), find j in [mid,hi)
        // where lower <= sums[j] - sums[i] <= upper
        int j1 = mid, j2 = mid;
        for (int i = lo; i < mid; i++) {
            while (j1 < hi && sums[j1] - sums[i] < lower) j1++;
            while (j2 < hi && sums[j2] - sums[i] <= upper) j2++;
            count += j2 - j1;
        }

        // Standard merge
        vector<long long> temp(hi - lo);
        int p = lo, q = mid, k = 0;
        while (p < mid && q < hi) {
            if (sums[p] <= sums[q]) temp[k++] = sums[p++];
            else temp[k++] = sums[q++];
        }
        while (p < mid) temp[k++] = sums[p++];
        while (q < hi) temp[k++] = sums[q++];
        for (int i = lo; i < hi; i++)
            sums[i] = temp[i - lo];
    }

public:
    int countRangeSum(vector<int>& nums, int lower, int upper) {
        this->lower = lower;
        this->upper = upper;
        count = 0;
        int n = nums.size();
        vector<long long> sums(n + 1, 0);
        for (int i = 0; i < n; i++)
            sums[i + 1] = sums[i] + nums[i];
        mergeSort(sums, 0, n + 1);
        return count;
    }
};
Why merge sort? After sorting left/right halves of prefix sums, we can use two pointers to count valid (j1, j2) windows in O(n) per level → total O(n log n).

08 — BIT vs Segment Tree

AspectBIT (Fenwick)Segment Tree
Code complexity ~10 lines ~40-60 lines
Memory O(n) — one array O(4n) — recursive, O(2n) — iterative
Point update O(log n) O(log n)
Range query O(log n) via prefix diff O(log n) direct
Range update O(log n) with diff array trick O(log n) with lazy propagation
Supported operations Invertible only (sum, xor) Any associative (min, max, gcd, sum)
Lazy propagation Not possible Supported
Constant factor Faster (cache-friendly) Slower (pointer chasing / recursion)

Interview tip: Start with BIT if the operation is invertible (sum/xor). Mention Segment Tree as upgrade path if interviewer asks for min/max or range updates.
