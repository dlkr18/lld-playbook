02 — Valid Parentheses (LC 20)

Push opening brackets, pop and match on closing brackets.

#include <stack>
#include <string>
using namespace std;

bool isValid(string s) {
    stack<char> st;
    for (char c : s) {
        if (c == '(' || c == '{' || c == '[') {
            st.push(c);
        } else {
            if (st.empty()) return false;
            char top = st.top(); st.pop();
            if (c == ')' && top != '(') return false;
            if (c == '}' && top != '{') return false;
            if (c == ']' && top != '[') return false;
        }
    }
    return st.empty();
}
Pattern: Push openers, pop & match closers. Final check: stack must be empty.

03 — Min Stack (LC 155)

Track minimum at each level using pairs: {value, current_min}.

#include <stack>
using namespace std;

class MinStack {
    stack<pair<int,int>> st; // {val, min_so_far}
public:
    void push(int val) {
        int curMin = st.empty() ? val : min(val, st.top().second);
        st.push({val, curMin});
    }
    void pop() { st.pop(); }
    int top() { return st.top().first; }
    int getMin() { return st.top().second; }
};
Alternative: Two separate stacks — one for values, one for mins. Same complexity, slightly more code.

04 — Monotonic Stack Template

Next Greater Element (right) — Decreasing Stack

// For each element, find the next greater element to its right
// Stack maintains DECREASING order (top is smallest)
vector<int> nextGreater(vector<int>& nums) {
    int n = nums.size();
    vector<int> res(n, -1);
    stack<int> st; // stores indices
    for (int i = 0; i < n; i++) {
        while (!st.empty() && nums[i] > nums[st.top()]) {
            res[st.top()] = nums[i];
            st.pop();
        }
        st.push(i);
    }
    return res;
}

Next Smaller Element (right) — Increasing Stack

// Stack maintains INCREASING order (top is largest)
vector<int> nextSmaller(vector<int>& nums) {
    int n = nums.size();
    vector<int> res(n, -1);
    stack<int> st;
    for (int i = 0; i < n; i++) {
        while (!st.empty() && nums[i] < nums[st.top()]) {
            res[st.top()] = nums[i];
            st.pop();
        }
        st.push(i);
    }
    return res;
}
Memory trick: "Pop when violated" — if you want next greater, maintain decreasing (pop when curr > top). If you want next smaller, maintain increasing (pop when curr < top).

05 — Next Greater Element I (LC 496)

Given nums1 (subset) and nums2, find next greater of each nums1[i] in nums2.

#include <vector>
#include <stack>
#include <unordered_map>
using namespace std;

vector<int> nextGreaterElement(vector<int>& nums1, vector<int>& nums2) {
    unordered_map<int,int> nge; // val -> next greater
    stack<int> st;
    for (int num : nums2) {
        while (!st.empty() && num > st.top()) {
            nge[st.top()] = num;
            st.pop();
        }
        st.push(num);
    }
    vector<int> res;
    for (int num : nums1) {
        res.push_back(nge.count(num) ? nge[num] : -1);
    }
    return res;
}

Build NGE map for all of nums2, then look up each element of nums1.

06 — Daily Temperatures (LC 739)

For each day, how many days until a warmer temperature? Classic monotonic decreasing stack on indices.

#include <vector>
#include <stack>
using namespace std;

vector<int> dailyTemperatures(vector<int>& temps) {
    int n = temps.size();
    vector<int> res(n, 0);
    stack<int> st; // indices of unresolved days
    for (int i = 0; i < n; i++) {
        while (!st.empty() && temps[i] > temps[st.top()]) {
            int prev = st.top(); st.pop();
            res[prev] = i - prev;
        }
        st.push(i);
    }
    return res;
}
Why decreasing? Stack holds days waiting for a warmer day. When we find a warmer day (curr > top), we pop and record the distance.

07 — Largest Rectangle in Histogram (LC 84)

Monotonic increasing stack. When we pop, the popped bar's width extends from the new top+1 to current index-1.

#include <vector>
#include <stack>
using namespace std;

int largestRectangleArea(vector<int>& heights) {
    stack<int> st;
    int maxArea = 0;
    int n = heights.size();

    for (int i = 0; i <= n; i++) {
        int curr = (i == n) ? 0 : heights[i]; // sentinel
        while (!st.empty() && curr < heights[st.top()]) {
            int h = heights[st.top()]; st.pop();
            int w = st.empty() ? i : i - st.top() - 1;
            maxArea = max(maxArea, h * w);
        }
        st.push(i);
    }
    return maxArea;
}

Why it works

  • Stack keeps indices in increasing height order
  • When curr < top, the top bar can't extend further right
  • Width = distance between new top and current index (exclusive on both sides)
  • Sentinel height 0 at the end forces all remaining bars to be popped

08 — Trapping Rain Water (LC 42)

Stack approach calculates water layer by layer (horizontal). Complementary to the two-pointer vertical approach.

#include <vector>
#include <stack>
using namespace std;

int trap(vector<int>& height) {
    stack<int> st;
    int water = 0;

    for (int i = 0; i < (int)height.size(); i++) {
        while (!st.empty() && height[i] > height[st.top()]) {
            int bottom = st.top(); st.pop();
            if (st.empty()) break;
            int w = i - st.top() - 1;
            int h = min(height[i], height[st.top()]) - height[bottom];
            water += w * h;
        }
        st.push(i);
    }
    return water;
}

How it works

  • Maintain a decreasing stack (like a valley detector)
  • When height[i] > height[top], we found a right wall
  • Pop the bottom, peek new top as left wall
  • Water trapped = width × min(left, right) - bottom

09 — Reverse Polish Notation (LC 150)

Operand stack: push numbers, pop two operands on operator, push result.

#include <vector>
#include <stack>
#include <string>
using namespace std;

int evalRPN(vector<string>& tokens) {
    stack<int> st;
    for (const string& t : tokens) {
        if (t == "+" || t == "-" || t == "*" || t == "/") {
            int b = st.top(); st.pop();
            int a = st.top(); st.pop();
            if (t == "+") st.push(a + b);
            else if (t == "-") st.push(a - b);
            else if (t == "*") st.push(a * b);
            else st.push(a / b); // truncate toward zero
        } else {
            st.push(stoi(t));
        }
    }
    return st.top();
}
Note: Order matters for subtraction/division — first popped is right operand (b), second is left (a).

10 — Asteroid Collision (LC 735)

Positive = moving right, negative = moving left. Collisions only happen when top is positive and current is negative.

#include <vector>
#include <stack>
using namespace std;

vector<int> asteroidCollision(vector<int>& asteroids) {
    vector<int> st; // use vector as stack for easy output

    for (int a : asteroids) {
        bool alive = true;
        while (alive && !st.empty() && a < 0 && st.back() > 0) {
            if (st.back() < -a) {
                st.pop_back(); // top explodes
            } else if (st.back() == -a) {
                st.pop_back(); // both explode
                alive = false;
            } else {
                alive = false; // current explodes
            }
        }
        if (alive) st.push_back(a);
    }
    return st;
}

Collision rules

  • Only collide when top > 0 and curr < 0 (moving toward each other)
  • Bigger asteroid survives, equal sizes both destroy
  • Use vector instead of stack for convenient final output
