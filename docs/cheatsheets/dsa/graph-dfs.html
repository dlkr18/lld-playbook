02 — DFS Templates

Adjacency List DFS (Recursive)

vector<vector<int>> adj; // adjacency list
vector<bool> visited;

void dfs(int u) {
    visited[u] = true;
    for (int v : adj[u]) {
        if (!visited[v])
            dfs(v);
    }
}

Iterative DFS (with Stack)

void dfs(int start) {
    stack<int> st;
    st.push(start);
    visited[start] = true;

    while (!st.empty()) {
        int u = st.top(); st.pop();
        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                st.push(v);
            }
        }
    }
}
When to use iterative? When recursion depth might cause stack overflow (e.g., linear chain of 105 nodes).

Building Adjacency List

// From edge list: edges[i] = {u, v}
vector<vector<int>> buildAdj(int n, vector<vector<int>>& edges) {
    vector<vector<int>> adj(n);
    for (auto& e : edges) {
        adj[e[0]].push_back(e[1]);
        adj[e[1]].push_back(e[0]); // omit for directed
    }
    return adj;
}

03 — Cycle Detection

Directed Graph — 3-Color DFS (LC 207)

Use three states: WHITE (unvisited), GRAY (in current path), BLACK (fully processed). A back edge (node → GRAY node) = cycle.

// 0 = white, 1 = gray, 2 = black
vector<int> color;

bool hasCycle(int u) {
    color[u] = 1; // gray = in progress
    for (int v : adj[u]) {
        if (color[v] == 1) return true;  // back edge → cycle
        if (color[v] == 0 && hasCycle(v)) return true;
    }
    color[u] = 2; // black = done
    return false;
}

bool canFinish(int n, vector<vector<int>>& prereqs) {
    adj.assign(n, {});
    color.assign(n, 0);
    for (auto& p : prereqs)
        adj[p[1]].push_back(p[0]);

    for (int i = 0; i < n; i++)
        if (color[i] == 0 && hasCycle(i)) return false;
    return true;
}

Undirected Graph — Parent Tracking

In undirected graphs, track the parent. Visiting a neighbor that is visited and not your parent = cycle.

bool dfsCycle(int u, int parent) {
    visited[u] = true;
    for (int v : adj[u]) {
        if (!visited[v]) {
            if (dfsCycle(v, u)) return true;
        } else if (v != parent) {
            return true; // back edge → cycle
        }
    }
    return false;
}

04 — Topological Sort

When to Use

DAG (Directed Acyclic Graph) only. Produces a linear ordering where for every edge u → v, u comes before v.

  • Course scheduling with prerequisites
  • Build system dependency resolution
  • Task ordering

Approach 1: Kahn's Algorithm (BFS / In-degree)

vector<int> topoSort(int n, vector<vector<int>>& adj) {
    vector<int> indeg(n, 0);
    for (int u = 0; u < n; u++)
        for (int v : adj[u])
            indeg[v]++;

    queue<int> q;
    for (int i = 0; i < n; i++)
        if (indeg[i] == 0) q.push(i);

    vector<int> order;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        order.push_back(u);
        for (int v : adj[u])
            if (--indeg[v] == 0)
                q.push(v);
    }

    if (order.size() != n) return {}; // cycle exists
    return order;
}
Cycle detection for free: if order.size() < n, the graph has a cycle.

Approach 2: DFS + Reverse Post-order

vector<int> order;
vector<bool> visited;

void dfs(int u) {
    visited[u] = true;
    for (int v : adj[u])
        if (!visited[v]) dfs(v);
    order.push_back(u); // post-order
}

vector<int> topoSortDFS(int n) {
    visited.assign(n, false);
    order.clear();
    for (int i = 0; i < n; i++)
        if (!visited[i]) dfs(i);
    reverse(order.begin(), order.end());
    return order;
}

Course Schedule II (LC 210)

vector<int> findOrder(int n, vector<vector<int>>& prereqs) {
    vector<vector<int>> adj(n);
    vector<int> indeg(n, 0);
    for (auto& p : prereqs) {
        adj[p[1]].push_back(p[0]);
        indeg[p[0]]++;
    }

    queue<int> q;
    for (int i = 0; i < n; i++)
        if (indeg[i] == 0) q.push(i);

    vector<int> order;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        order.push_back(u);
        for (int v : adj[u])
            if (--indeg[v] == 0) q.push(v);
    }
    return order.size() == n ? order : vector<int>{};
}

05 — Connected Components

Number of Provinces (LC 547)

Given an adjacency matrix, count connected components. DFS from each unvisited node = one component.

int findCircleNum(vector<vector<int>>& isConnected) {
    int n = isConnected.size(), count = 0;
    vector<bool> visited(n, false);

    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            dfs(isConnected, visited, i);
            count++;
        }
    }
    return count;
}

void dfs(vector<vector<int>>& g, vector<bool>& vis, int u) {
    vis[u] = true;
    for (int v = 0; v < g.size(); v++)
        if (g[u][v] && !vis[v])
            dfs(g, vis, v);
}

06 — Strongly Connected Components

What is SCC?

In a directed graph, an SCC is a maximal set of vertices where every vertex is reachable from every other vertex in the set.

Kosaraju's Algorithm

  1. DFS on original graph, push to stack in finish order
  2. Build transposed graph (reverse all edges)
  3. Pop from stack, DFS on transposed graph — each DFS = one SCC
vector<vector<int>> adj, radj;
vector<bool> visited;
stack<int> finish;

void dfs1(int u) {
    visited[u] = true;
    for (int v : adj[u])
        if (!visited[v]) dfs1(v);
    finish.push(u);
}

void dfs2(int u, vector<int>& comp) {
    visited[u] = true;
    comp.push_back(u);
    for (int v : radj[u])
        if (!visited[v]) dfs2(v, comp);
}

vector<vector<int>> kosaraju(int n) {
    visited.assign(n, false);
    for (int i = 0; i < n; i++)
        if (!visited[i]) dfs1(i);

    visited.assign(n, false);
    vector<vector<int>> sccs;
    while (!finish.empty()) {
        int u = finish.top(); finish.pop();
        if (!visited[u]) {
            vector<int> comp;
            dfs2(u, comp);
            sccs.push_back(comp);
        }
    }
    return sccs;
}

Tarjan's Algorithm (Bridges / Articulation)

Finds bridges (Critical Connections, LC 1192) using DFS discovery time and low-link values.

int timer = 0;
vector<int> disc, low;
vector<vector<int>> bridges;

void dfs(int u, int parent) {
    disc[u] = low[u] = timer++;
    for (int v : adj[u]) {
        if (disc[v] == -1) {
            dfs(v, u);
            low[u] = min(low[u], low[v]);
            if (low[v] > disc[u])
                bridges.push_back({u, v});
        } else if (v != parent) {
            low[u] = min(low[u], disc[v]);
        }
    }
}

vector<vector<int>> criticalConnections(int n, vector<vector<int>>& conns) {
    adj.assign(n, {});
    disc.assign(n, -1);
    low.assign(n, -1);
    for (auto& e : conns) {
        adj[e[0]].push_back(e[1]);
        adj[e[1]].push_back(e[0]);
    }
    dfs(0, -1);
    return bridges;
}
Bridge condition: edge u—v is a bridge if low[v] > disc[u] (no back edge from v's subtree can reach u or above).

Articulation Points (Cut Vertices)

A vertex whose removal disconnects the graph. Same disc/low idea, different condition.

vector<bool> isAP;

void dfsAP(int u, int parent) {
    disc[u] = low[u] = timer++;
    int children = 0;

    for (int v : adj[u]) {
        if (disc[v] == -1) {
            children++;
            dfsAP(v, u);
            low[u] = min(low[u], low[v]);

            // Non-root: u is AP if child can't reach above u
            if (parent != -1 && low[v] >= disc[u])
                isAP[u] = true;

            // Root: AP only if it has 2+ DFS children
            if (parent == -1 && children > 1)
                isAP[u] = true;
        } else if (v != parent) {
            low[u] = min(low[u], disc[v]);
        }
    }
}

07 — Bipartite Check

Is Graph Bipartite? (LC 785)

Try to 2-color the graph. If any neighbor has the same color, it's not bipartite.

bool isBipartite(vector<vector<int>>& graph) {
    int n = graph.size();
    vector<int> color(n, -1);

    for (int i = 0; i < n; i++) {
        if (color[i] != -1) continue;
        queue<int> q;
        q.push(i);
        color[i] = 0;

        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : graph[u]) {
                if (color[v] == -1) {
                    color[v] = 1 - color[u];
                    q.push(v);
                } else if (color[v] == color[u]) {
                    return false;
                }
            }
        }
    }
    return true;
}

Key Insight

  • A graph is bipartite ⇔ it has no odd-length cycle
  • Works for disconnected graphs (loop over all nodes)
  • BFS or DFS both work — BFS is cleaner for coloring
