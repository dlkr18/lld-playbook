Interview tip: “Which sort would you use?” = know the trade-offs. Merge for stability/guarantees, Quick for avg speed/in-place, Counting/Radix when range is bounded.

02 — Merge Sort Template

Divide & conquer. Split in half, sort each, merge two sorted halves. Stable, O(n) extra space.

void merge(vector<int>& arr, int lo, int mid, int hi) {
    vector<int> tmp(hi - lo + 1);
    int i = lo, j = mid + 1, k = 0;
    while (i <= mid && j <= hi) {
        if (arr[i] <= arr[j]) tmp[k++] = arr[i++];  // <= keeps stability
        else                  tmp[k++] = arr[j++];
    }
    while (i <= mid) tmp[k++] = arr[i++];
    while (j <= hi)  tmp[k++] = arr[j++];
    for (int p = 0; p < k; p++) arr[lo + p] = tmp[p];
}

void mergeSort(vector<int>& arr, int lo, int hi) {
    if (lo >= hi) return;
    int mid = lo + (hi - lo) / 2;
    mergeSort(arr, lo, mid);
    mergeSort(arr, mid + 1, hi);
    merge(arr, lo, mid, hi);
}
Key insight: The merge step is where all the magic happens. It's also the building block for count-inversions, count-smaller-after-self, etc.

03 — Quick Sort (Lomuto Partition)

Pick pivot (last element), partition so everything ≤ pivot is on the left. Recurse on both sides.

int partition(vector<int>& arr, int lo, int hi) {
    int pivot = arr[hi];
    int i = lo;  // i = boundary of “≤ pivot” region
    for (int j = lo; j < hi; j++) {
        if (arr[j] <= pivot) {
            swap(arr[i], arr[j]);
            i++;
        }
    }
    swap(arr[i], arr[hi]);  // place pivot in correct position
    return i;
}

void quickSort(vector<int>& arr, int lo, int hi) {
    if (lo >= hi) return;
    int p = partition(arr, lo, hi);
    quickSort(arr, lo, p - 1);
    quickSort(arr, p + 1, hi);
}
Randomized pivot: To avoid O(n²) on sorted input, swap arr[lo + rand() % (hi-lo+1)] with arr[hi] before partitioning.

04 — Quick Select — Kth Largest (LC 215)

Find the kth largest element in O(n) average without fully sorting. Uses the same Lomuto partition — only recurse into the half that contains the answer.

#include <vector>
#include <cstdlib>
using namespace std;

class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        int target = nums.size() - k;  // kth largest = (n-k)th smallest
        int lo = 0, hi = nums.size() - 1;
        while (lo < hi) {
            int p = partition(nums, lo, hi);
            if (p == target) return nums[p];
            else if (p < target) lo = p + 1;
            else hi = p - 1;
        }
        return nums[lo];
    }

private:
    int partition(vector<int>& arr, int lo, int hi) {
        // randomized pivot to avoid O(n²) worst case
        int randIdx = lo + rand() % (hi - lo + 1);
        swap(arr[randIdx], arr[hi]);

        int pivot = arr[hi], i = lo;
        for (int j = lo; j < hi; j++) {
            if (arr[j] <= pivot) {
                swap(arr[i], arr[j]);
                i++;
            }
        }
        swap(arr[i], arr[hi]);
        return i;
    }
};
  • Time: O(n) average, O(n²) worst (random pivot makes worst case astronomically unlikely)
  • Space: O(1) — in-place, iterative
  • Alternative: Use a max-heap of size k for O(n log k) guaranteed

05 — Sort Colors / Dutch National Flag (LC 75)

3-way partition in one pass, O(1) space. Three pointers: lo (next 0 position), mid (current), hi (next 2 position).

#include <vector>
using namespace std;

class Solution {
public:
    void sortColors(vector<int>& nums) {
        int lo = 0, mid = 0, hi = nums.size() - 1;
        while (mid <= hi) {
            if (nums[mid] == 0) {
                swap(nums[lo], nums[mid]);
                lo++; mid++;
            } else if (nums[mid] == 1) {
                mid++;
            } else {  // nums[mid] == 2
                swap(nums[mid], nums[hi]);
                hi--;   // don’t advance mid — swapped value unknown
            }
        }
    }
};
Invariant: [0..lo-1] = all 0s, [lo..mid-1] = all 1s, [hi+1..n-1] = all 2s, [mid..hi] = unprocessed.
  • Time: O(n) single pass
  • Space: O(1)
  • Gotcha: Don’t increment mid after swapping with hi — the swapped element hasn’t been examined yet

06 — Count Inversions (Merge Sort Application)

Count pairs (i, j) where i < j but arr[i] > arr[j]. Classic merge sort modification — count cross-inversions during merge.

#include <vector>
using namespace std;

long long mergeCount(vector<int>& arr, int lo, int mid, int hi) {
    vector<int> tmp(hi - lo + 1);
    int i = lo, j = mid + 1, k = 0;
    long long inv = 0;
    while (i <= mid && j <= hi) {
        if (arr[i] <= arr[j]) {
            tmp[k++] = arr[i++];
        } else {
            tmp[k++] = arr[j++];
            inv += (mid - i + 1);  // all remaining in left half are inversions
        }
    }
    while (i <= mid) tmp[k++] = arr[i++];
    while (j <= hi)  tmp[k++] = arr[j++];
    for (int p = 0; p < k; p++) arr[lo + p] = tmp[p];
    return inv;
}

long long countInversions(vector<int>& arr, int lo, int hi) {
    if (lo >= hi) return 0;
    int mid = lo + (hi - lo) / 2;
    long long inv = 0;
    inv += countInversions(arr, lo, mid);
    inv += countInversions(arr, mid + 1, hi);
    inv += mergeCount(arr, lo, mid, hi);
    return inv;
}

// Usage: countInversions(arr, 0, arr.size() - 1)
  • Time: O(n log n)
  • Key idea: When right element is smaller, every remaining element in the left half forms an inversion with it → inv += (mid - i + 1)
  • Same pattern: LC 315 (Count Smaller After Self), LC 493 (Reverse Pairs)

07 — Counting Sort Template

When values are in a known, small range [0, k). Count occurrences, compute prefix sums, place elements. Stable.

#include <vector>
using namespace std;

vector<int> countingSort(vector<int>& arr, int maxVal) {
    vector<int> count(maxVal + 1, 0);
    for (int x : arr) count[x]++;

    // prefix sums: count[i] = number of elements ≤ i
    for (int i = 1; i <= maxVal; i++)
        count[i] += count[i - 1];

    vector<int> output(arr.size());
    // traverse right-to-left for stability
    for (int i = arr.size() - 1; i >= 0; i--) {
        output[count[arr[i]] - 1] = arr[i];
        count[arr[i]]--;
    }
    return output;
}
  • Time: O(n + k), Space: O(n + k)
  • Use when: k = O(n), e.g. sorting ages, characters, small enums
  • Simplified version: If you only need sorted output (not stability), just iterate count array and emit values

08 — Bucket Sort Template

Distribute elements into buckets, sort each bucket individually. Best for uniformly distributed data.

#include <vector>
#include <algorithm>
using namespace std;

void bucketSort(vector<float>& arr) {
    int n = arr.size();
    vector<vector<float>> buckets(n);

    // distribute into buckets [0, 1)
    for (float x : arr) {
        int idx = (int)(n * x);  // assumes x in [0, 1)
        buckets[idx].push_back(x);
    }

    // sort individual buckets (insertion sort is fine for small buckets)
    for (auto& b : buckets)
        sort(b.begin(), b.end());

    // concatenate
    int idx = 0;
    for (auto& b : buckets)
        for (float x : b)
            arr[idx++] = x;
}
  • Time: O(n + k) average if uniform, O(n²) worst case
  • Use when: floats in [0,1), or can normalize to bounded range
  • For integers: Choose bucket size = (max - min) / n + 1

09 — Sort an Array (LC 912)

Full merge sort implementation that passes LC 912. Clean, interview-ready.

#include <vector>
using namespace std;

class Solution {
public:
    vector<int> sortArray(vector<int>& nums) {
        mergeSort(nums, 0, nums.size() - 1);
        return nums;
    }

private:
    void mergeSort(vector<int>& a, int lo, int hi) {
        if (lo >= hi) return;
        int mid = lo + (hi - lo) / 2;
        mergeSort(a, lo, mid);
        mergeSort(a, mid + 1, hi);
        merge(a, lo, mid, hi);
    }

    void merge(vector<int>& a, int lo, int mid, int hi) {
        vector<int> tmp(hi - lo + 1);
        int i = lo, j = mid + 1, k = 0;
        while (i <= mid && j <= hi) {
            if (a[i] <= a[j]) tmp[k++] = a[i++];
            else               tmp[k++] = a[j++];
        }
        while (i <= mid) tmp[k++] = a[i++];
        while (j <= hi)  tmp[k++] = a[j++];
        for (int p = 0; p < k; p++) a[lo + p] = tmp[p];
    }
};
Why merge sort for LC 912? Quick sort TLEs on sorted/reverse-sorted inputs unless you randomize pivot. Merge sort is guaranteed O(n log n).
