01 — Pattern Guide

When to Use Trie vs HashMap vs Sorting

ScenarioBest StructureWhy
Exact word lookup only unordered_set O(1) average, simpler code
Prefix matching (startsWith, autocomplete) Trie O(L) prefix check; HashMap can't enumerate by prefix
Wildcard characters (. matches any) Trie + DFS Branch at wildcard, prune non-matching paths
Multi-word search on a grid Trie + Backtracking One DFS pass finds all words vs repeating per word
Longest common prefix of N strings Trie or vertical scan Trie: O(S) build + walk single-child chain
Replace words with shortest prefix in dict Trie Early termination at first isEnd node
Lexicographic sort Trie DFS or sort() DFS of trie yields sorted order naturally
Rule of thumb: If the problem mentions "prefix", "starts with", "wildcard", or "find all words matching" — think Trie first. If it's just membership testing, use a HashSet.

02 — Trie Node Structure + Template

Standard TrieNode

struct TrieNode {
    TrieNode* children[26] = {};
    bool isEnd = false;
};

Each node has 26 child pointers (lowercase a-z). isEnd marks a complete word. For larger alphabets, use unordered_map<char, TrieNode*>.

Insert / Search / StartsWith Template

class Trie {
    TrieNode* root;
public:
    Trie() : root(new TrieNode()) {}

    void insert(string& word) {
        TrieNode* node = root;
        for (char c : word) {
            int idx = c - 'a';
            if (!node->children[idx])
                node->children[idx] = new TrieNode();
            node = node->children[idx];
        }
        node->isEnd = true;
    }

    bool search(string& word) {
        TrieNode* node = traverse(word);
        return node && node->isEnd;
    }

    bool startsWith(string& prefix) {
        return traverse(prefix) != nullptr;
    }

private:
    TrieNode* traverse(string& s) {
        TrieNode* node = root;
        for (char c : s) {
            int idx = c - 'a';
            if (!node->children[idx]) return nullptr;
            node = node->children[idx];
        }
        return node;
    }
};

03 — Implement Trie (LC 208)

Problem

Implement a trie with insert, search, and startsWith methods.

Complete Solution

struct TrieNode {
    TrieNode* children[26] = {};
    bool isEnd = false;
};

class Trie {
    TrieNode* root;
public:
    Trie() { root = new TrieNode(); }

    void insert(string word) {
        TrieNode* cur = root;
        for (char c : word) {
            int i = c - 'a';
            if (!cur->children[i])
                cur->children[i] = new TrieNode();
            cur = cur->children[i];
        }
        cur->isEnd = true;
    }

    bool search(string word) {
        TrieNode* node = find(word);
        return node != nullptr && node->isEnd;
    }

    bool startsWith(string prefix) {
        return find(prefix) != nullptr;
    }

private:
    TrieNode* find(string& s) {
        TrieNode* cur = root;
        for (char c : s) {
            int i = c - 'a';
            if (!cur->children[i]) return nullptr;
            cur = cur->children[i];
        }
        return cur;
    }
};

Complexity

OperationTimeSpace
insert(word)O(L)O(L) worst case new nodes
search(word)O(L)O(1)
startsWith(prefix)O(L)O(1)

L = length of word/prefix. Total space: O(N*L) worst case for N words.

04 — Word Search II (LC 212)

Core Idea

Build a trie from the word list. DFS from every cell on the board, walking the trie simultaneously. When isEnd is reached, a word is found. Prune found words to avoid duplicates.

Key insight: Without trie, you'd run DFS for each word separately — O(words * m*n * 4^L). With trie, one DFS pass from each cell checks all words simultaneously.

Complete Solution

struct TrieNode {
    TrieNode* children[26] = {};
    string word = "";  // store word at leaf for easy retrieval
};

class Solution {
    vector<string> result;
    int m, n;

    void dfs(vector<vector<char>>& board, int r, int c, TrieNode* node) {
        if (r < 0 || r >= m || c < 0 || c >= n) return;
        char ch = board[r][c];
        if (ch == '#' || !node->children[ch - 'a']) return;

        node = node->children[ch - 'a'];

        if (!node->word.empty()) {
            result.push_back(node->word);
            node->word = "";  // de-duplicate
        }

        board[r][c] = '#';  // mark visited
        dfs(board, r + 1, c, node);
        dfs(board, r - 1, c, node);
        dfs(board, r, c + 1, node);
        dfs(board, r, c - 1, node);
        board[r][c] = ch;  // restore

        // optimization: prune empty branches
        bool hasChild = false;
        for (int i = 0; i < 26; i++) {
            if (node->children[i]) { hasChild = true; break; }
        }
        if (!hasChild && node->word.empty()) {
            // remove this node from parent (optional perf boost)
        }
    }

public:
    vector<string> findWords(vector<vector<char>>& board,
                              vector<string>& words) {
        // Build trie
        TrieNode* root = new TrieNode();
        for (string& w : words) {
            TrieNode* node = root;
            for (char c : w) {
                int i = c - 'a';
                if (!node->children[i])
                    node->children[i] = new TrieNode();
                node = node->children[i];
            }
            node->word = w;
        }

        m = board.size(); n = board[0].size();
        for (int r = 0; r < m; r++)
            for (int c = 0; c < n; c++)
                dfs(board, r, c, root);

        return result;
    }
};

Complexity

AspectValue
TimeO(m * n * 4^L) where L = max word length, but trie prunes aggressively
SpaceO(W * L) for trie + O(L) recursion stack

05 — Design Add and Search Words (LC 211)

Core Idea

search("b.d") — the . matches any character. Use DFS: at a ., branch into all 26 children that exist.

Complete Solution

struct TrieNode {
    TrieNode* children[26] = {};
    bool isEnd = false;
};

class WordDictionary {
    TrieNode* root;

    bool dfs(TrieNode* node, string& word, int idx) {
        if (!node) return false;
        if (idx == word.size()) return node->isEnd;

        char c = word[idx];
        if (c == '.') {
            // wildcard: try all existing children
            for (int i = 0; i < 26; i++) {
                if (dfs(node->children[i], word, idx + 1))
                    return true;
            }
            return false;
        } else {
            return dfs(node->children[c - 'a'], word, idx + 1);
        }
    }

public:
    WordDictionary() { root = new TrieNode(); }

    void addWord(string word) {
        TrieNode* cur = root;
        for (char c : word) {
            int i = c - 'a';
            if (!cur->children[i])
                cur->children[i] = new TrieNode();
            cur = cur->children[i];
        }
        cur->isEnd = true;
    }

    bool search(string word) {
        return dfs(root, word, 0);
    }
};

Complexity

OperationTimeNotes
addWordO(L)Same as standard insert
search (no wildcard)O(L)Direct traversal
search (with .)O(26^D * L)D = number of dots; worst case exponential

06 — Longest Common Prefix (LC 14)

Core Idea

Insert all strings into a trie. Walk from root, following the only child at each level, until you hit a node with multiple children or isEnd == true. The path traversed = longest common prefix.

Trie Approach

struct TrieNode {
    TrieNode* children[26] = {};
    bool isEnd = false;
    int childCount = 0;
};

string longestCommonPrefix(vector<string>& strs) {
    if (strs.empty()) return "";

    // Build trie
    TrieNode* root = new TrieNode();
    for (string& s : strs) {
        TrieNode* cur = root;
        for (char c : s) {
            int i = c - 'a';
            if (!cur->children[i]) {
                cur->children[i] = new TrieNode();
                cur->childCount++;
            }
            cur = cur->children[i];
        }
        cur->isEnd = true;
    }

    // Walk single-child chain
    string prefix = "";
    TrieNode* cur = root;
    while (cur->childCount == 1 && !cur->isEnd) {
        for (int i = 0; i < 26; i++) {
            if (cur->children[i]) {
                prefix += (char)('a' + i);
                cur = cur->children[i];
                break;
            }
        }
    }
    return prefix;
}

Simpler Vertical Scan (No Trie)

string longestCommonPrefix(vector<string>& strs) {
    if (strs.empty()) return "";
    for (int i = 0; i < strs[0].size(); i++) {
        char c = strs[0][i];
        for (int j = 1; j < strs.size(); j++) {
            if (i >= strs[j].size() || strs[j][i] != c)
                return strs[0].substr(0, i);
        }
    }
    return strs[0];
}

Both are O(S) where S = sum of all characters. Trie is better when you need the structure for other queries too.

07 — Replace Words (LC 648)

Problem

Given a dictionary of roots and a sentence, replace every word in the sentence with its shortest root that is a prefix. E.g., dictionary = ["cat","bat","rat"], sentence = "the cattle was rattled by the battery" → "the cat was rat by the bat".

Complete Solution

struct TrieNode {
    TrieNode* children[26] = {};
    bool isEnd = false;
};

class Solution {
public:
    string replaceWords(vector<string>& dictionary, string sentence) {
        // Build trie from dictionary
        TrieNode* root = new TrieNode();
        for (string& word : dictionary) {
            TrieNode* cur = root;
            for (char c : word) {
                int i = c - 'a';
                if (!cur->children[i])
                    cur->children[i] = new TrieNode();
                cur = cur->children[i];
            }
            cur->isEnd = true;
        }

        // Process each word in sentence
        string result, word;
        istringstream iss(sentence);
        bool first = true;

        while (iss >> word) {
            if (!first) result += ' ';
            first = false;

            TrieNode* cur = root;
            string prefix = "";
            bool found = false;

            for (char c : word) {
                int i = c - 'a';
                if (!cur->children[i]) break;
                prefix += c;
                cur = cur->children[i];
                if (cur->isEnd) { found = true; break; }
            }

            result += found ? prefix : word;
        }
        return result;
    }
};

Complexity

AspectValue
Build trieO(D) where D = total chars in dictionary
Process sentenceO(S) where S = total chars in sentence
SpaceO(D) for trie
Why trie beats HashSet here: With a HashSet, you'd check every possible prefix of each word against the set — O(L²) per word. Trie does it in O(L) with early termination at first isEnd.
