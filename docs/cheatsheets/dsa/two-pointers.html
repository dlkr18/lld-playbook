02 — Two Sum II Sorted (LC 167)

Given a 1-indexed sorted array, find two numbers that add up to target. Return indices.

vector<int> twoSum(vector<int>& nums, int target) {
    int l = 0, r = nums.size() - 1;
    while (l < r) {
        int sum = nums[l] + nums[r];
        if (sum == target) return {l + 1, r + 1};
        else if (sum < target) l++;
        else r--;
    }
    return {};
}
Why it works: Array is sorted. If sum is too small, move left pointer right to increase it. If too large, move right pointer left.

03 — 3Sum (LC 15)

Find all unique triplets that sum to zero. Sort + fix one element + two-pointer on the rest.

vector<vector<int>> threeSum(vector<int>& nums) {
    vector<vector<int>> res;
    sort(nums.begin(), nums.end());
    int n = nums.size();

    for (int i = 0; i < n - 2; i++) {
        if (i > 0 && nums[i] == nums[i - 1]) continue;  // skip dup
        int l = i + 1, r = n - 1;
        while (l < r) {
            int sum = nums[i] + nums[l] + nums[r];
            if (sum == 0) {
                res.push_back({nums[i], nums[l], nums[r]});
                while (l < r && nums[l] == nums[l + 1]) l++;  // skip dup
                while (l < r && nums[r] == nums[r - 1]) r--;  // skip dup
                l++; r--;
            } else if (sum < 0) l++;
            else r--;
        }
    }
    return res;
}
Dedup trick: Skip duplicate values for both the outer loop nums[i]==nums[i-1] and inner pointers after finding a triplet.

04 — Container With Most Water (LC 11)

Find two lines that form the container holding the most water. Always move the shorter pointer inward.

int maxArea(vector<int>& height) {
    int l = 0, r = height.size() - 1;
    int ans = 0;
    while (l < r) {
        int area = min(height[l], height[r]) * (r - l);
        ans = max(ans, area);
        if (height[l] < height[r]) l++;
        else r--;
    }
    return ans;
}
Greedy insight: Moving the shorter line is the only way to potentially find a taller line. Moving the taller line can never increase the area (width shrinks, height stays bounded by the shorter).

05 — Trapping Rain Water (LC 42)

Compute trapped water between bars. Two-pointer avoids the need for prefix/suffix max arrays.

int trap(vector<int>& height) {
    int l = 0, r = height.size() - 1;
    int lmax = 0, rmax = 0, water = 0;
    while (l < r) {
        if (height[l] < height[r]) {
            if (height[l] >= lmax) lmax = height[l];
            else water += lmax - height[l];
            l++;
        } else {
            if (height[r] >= rmax) rmax = height[r];
            else water += rmax - height[r];
            r--;
        }
    }
    return water;
}
Key insight: Water at position i = min(leftMax, rightMax) - height[i]. We process from the side with the smaller max, guaranteeing the other side has a wall at least as tall.

06 — Fixed-Size Sliding Window Template

For any problem asking about k consecutive elements — max sum, average, count, etc.

// Template: fixed window of size k
int fixedWindow(vector<int>& nums, int k) {
    int n = nums.size();
    int windowSum = 0, ans = INT_MIN;

    for (int i = 0; i < n; i++) {
        windowSum += nums[i];             // expand: add right
        if (i >= k) {
            windowSum -= nums[i - k];     // shrink: remove left
        }
        if (i >= k - 1) {
            ans = max(ans, windowSum);   // window is full, record
        }
    }
    return ans;
}
  • Window becomes valid at i == k-1
  • Remove element leaving the window: nums[i - k]
  • O(n) time, O(1) space

07 — Maximum Average Subarray (LC 643)

Find the contiguous subarray of length k with the maximum average.

double findMaxAverage(vector<int>& nums, int k) {
    double windowSum = 0;
    for (int i = 0; i < k; i++)
        windowSum += nums[i];

    double maxSum = windowSum;
    for (int i = k; i < nums.size(); i++) {
        windowSum += nums[i] - nums[i - k];
        maxSum = max(maxSum, windowSum);
    }
    return maxSum / k;
}

Classic fixed window. Compute initial window sum, then slide: add new right, remove old left.

08 — Variable-Size Sliding Window Template

For problems asking longest/shortest subarray satisfying a condition.

// Template: variable window (shrink-expand)
int variableWindow(vector<int>& nums) {
    int l = 0, ans = 0;

    for (int r = 0; r < nums.size(); r++) {
        // EXPAND: add nums[r] to window state

        while (/* window is INVALID */) {
            // SHRINK: remove nums[l] from window state
            l++;
        }
        // Window [l..r] is valid here
        ans = max(ans, r - l + 1);  // longest
        // ans = min(ans, r - l + 1);  // shortest
    }
    return ans;
}
Pattern: Right pointer always advances (expand). Left pointer advances only when constraint is violated (shrink). Each element enters and leaves the window at most once → O(n).

09 — Minimum Window Substring (LC 76)

Find the smallest window in s containing all characters of t. Variable window + frequency map.

string minWindow(string s, string t) {
    unordered_map<char, int> need, have;
    for (char c : t) need[c]++;

    int required = need.size();  // distinct chars needed
    int formed = 0;              // distinct chars satisfied
    int l = 0, bestL = 0, bestLen = INT_MAX;

    for (int r = 0; r < s.size(); r++) {
        char c = s[r];
        have[c]++;
        if (need.count(c) && have[c] == need[c])
            formed++;

        while (formed == required) {      // window valid → shrink
            if (r - l + 1 < bestLen) {
                bestLen = r - l + 1;
                bestL = l;
            }
            char d = s[l];
            have[d]--;
            if (need.count(d) && have[d] < need[d])
                formed--;
            l++;
        }
    }
    return bestLen == INT_MAX ? "" : s.substr(bestL, bestLen);
}
Optimization: Track formed count instead of comparing entire maps each iteration. Only O(1) work per character.

10 — Longest Substring Without Repeating (LC 3)

Find the length of the longest substring with all unique characters.

int lengthOfLongestSubstring(string s) {
    unordered_map<char, int> lastSeen;
    int l = 0, ans = 0;

    for (int r = 0; r < s.size(); r++) {
        if (lastSeen.count(s[r]) && lastSeen[s[r]] >= l) {
            l = lastSeen[s[r]] + 1;  // jump left past the dup
        }
        lastSeen[s[r]] = r;
        ans = max(ans, r - l + 1);
    }
    return ans;
}
Trick: Instead of shrinking one-by-one, jump l directly to lastSeen[char]+1. Still O(n) but fewer iterations in practice.
