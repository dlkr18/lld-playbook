02 — KMP Algorithm

Build LPS (Longest Proper Prefix which is also Suffix)

vector<int> buildLPS(const string& pat) {
    int m = pat.size();
    vector<int> lps(m, 0);
    int len = 0, i = 1;
    while (i < m) {
        if (pat[i] == pat[len]) {
            lps[i++] = ++len;
        } else if (len > 0) {
            len = lps[len - 1];  // don't increment i
        } else {
            lps[i++] = 0;
        }
    }
    return lps;
}

KMP Search

vector<int> kmpSearch(const string& text, const string& pat) {
    int n = text.size(), m = pat.size();
    vector<int> lps = buildLPS(pat);
    vector<int> result;
    int i = 0, j = 0;
    while (i < n) {
        if (text[i] == pat[j]) {
            i++; j++;
        }
        if (j == m) {
            result.push_back(i - j);  // match found at index i-j
            j = lps[j - 1];
        } else if (i < n && text[i] != pat[j]) {
            if (j > 0) j = lps[j - 1];
            else i++;
        }
    }
    return result;
}
Time: O(n + m) — never re-scan characters. LPS array tells us the longest prefix of pattern that is also a suffix, so we skip ahead on mismatch.

03 — Rabin-Karp Algorithm

Rolling hash for O(n) average-case pattern matching. Good for multi-pattern search.

vector<int> rabinKarp(const string& text, const string& pat) {
    int n = text.size(), m = pat.size();
    if (m > n) return {};
    vector<int> result;

    const long long BASE = 31, MOD = 1e9 + 7;
    long long patHash = 0, txtHash = 0, power = 1;

    // compute hash of pattern and first window
    for (int i = 0; i < m; i++) {
        patHash = (patHash + (long long)(pat[i] - 'a' + 1) * power) % MOD;
        txtHash = (txtHash + (long long)(text[i] - 'a' + 1) * power) % MOD;
        if (i < m - 1) power = (power * BASE) % MOD;
    }

    for (int i = 0; i <= n - m; i++) {
        if (patHash == txtHash) {
            // verify to avoid hash collision
            if (text.substr(i, m) == pat)
                result.push_back(i);
        }
        if (i < n - m) {
            // slide window: remove left char, add right char
            txtHash = (txtHash - (long long)(text[i] - 'a' + 1) + MOD) % MOD;
            txtHash = (txtHash / BASE + (long long)(text[i + m] - 'a' + 1) * power) % MOD;
        }
    }
    return result;
}

// Cleaner alternative using modular inverse for rolling:
vector<int> rabinKarpV2(const string& text, const string& pat) {
    int n = text.size(), m = pat.size();
    if (m > n) return {};
    vector<int> result;

    const long long BASE = 31, MOD = 1e9 + 7;
    long long patHash = 0, txtHash = 0;
    long long highPow = 1;  // BASE^(m-1)

    for (int i = 0; i < m - 1; i++)
        highPow = (highPow * BASE) % MOD;

    for (int i = 0; i < m; i++) {
        patHash = (patHash * BASE + (pat[i] - 'a' + 1)) % MOD;
        txtHash = (txtHash * BASE + (text[i] - 'a' + 1)) % MOD;
    }

    for (int i = 0; i <= n - m; i++) {
        if (patHash == txtHash && text.substr(i, m) == pat)
            result.push_back(i);
        if (i < n - m) {
            txtHash = (txtHash - (long long)(text[i] - 'a' + 1) * highPow % MOD + MOD) % MOD;
            txtHash = (txtHash * BASE + (text[i + m] - 'a' + 1)) % MOD;
        }
    }
    return result;
}
Average O(n + m), worst O(nm) due to hash collisions. Use double hashing for contests.

04 — Z-Algorithm

z[i] = length of longest substring starting at i which is also a prefix of the string.

vector<int> zFunction(const string& s) {
    int n = s.size();
    vector<int> z(n, 0);
    int l = 0, r = 0;
    for (int i = 1; i < n; i++) {
        if (i < r)
            z[i] = min(r - i, z[i - l]);
        while (i + z[i] < n && s[z[i]] == s[i + z[i]])
            z[i]++;
        if (i + z[i] > r) {
            l = i;
            r = i + z[i];
        }
    }
    return z;
}

Pattern Matching with Z-Algorithm

vector<int> zSearch(const string& text, const string& pat) {
    string combined = pat + "$" + text;  // $ never appears in either
    vector<int> z = zFunction(combined);
    vector<int> result;
    int m = pat.size();
    for (int i = m + 1; i < (int)combined.size(); i++) {
        if (z[i] == m)
            result.push_back(i - m - 1);
    }
    return result;
}
Time: O(n + m). Build "pattern$text", any position with z[i] == len(pattern) is a match.

05 — Longest Palindromic Substring (LC 5)

Expand Around Center — O(n²)

string longestPalindrome(string s) {
    int n = s.size(), start = 0, maxLen = 1;

    auto expand = [&](int l, int r) {
        while (l >= 0 && r < n && s[l] == s[r]) {
            if (r - l + 1 > maxLen) {
                start = l;
                maxLen = r - l + 1;
            }
            l--; r++;
        }
    };

    for (int i = 0; i < n; i++) {
        expand(i, i);      // odd-length palindromes
        expand(i, i + 1);  // even-length palindromes
    }
    return s.substr(start, maxLen);
}
Manacher's Algorithm solves this in O(n) by reusing previously computed palindrome radii — rarely asked to implement but good to mention as a follow-up.

06 — Longest Palindromic Subsequence (LC 516)

dp[i][j] = LPS length of s[i..j]

int longestPalindromeSubseq(string s) {
    int n = s.size();
    vector<vector<int>> dp(n, vector<int>(n, 0));

    // base case: single chars are palindromes of length 1
    for (int i = 0; i < n; i++)
        dp[i][i] = 1;

    // fill bottom-up by length
    for (int len = 2; len <= n; len++) {
        for (int i = 0; i <= n - len; i++) {
            int j = i + len - 1;
            if (s[i] == s[j]) {
                dp[i][j] = dp[i + 1][j - 1] + 2;
            } else {
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1]);
            }
        }
    }
    return dp[0][n - 1];
}

// Space-optimized O(n) using 1D DP:
int longestPalindromeSubseqOpt(string s) {
    int n = s.size();
    vector<int> dp(n, 1), prev(n, 0);

    for (int len = 2; len <= n; len++) {
        vector<int> cur(n, 0);
        for (int i = 0; i <= n - len; i++) {
            int j = i + len - 1;
            if (s[i] == s[j])
                cur[i] = prev[i + 1] + 2;
            else
                cur[i] = max(dp[i + 1], dp[i]);
        }
        prev = dp;
        dp = cur;
    }
    return dp[0];
}
Time: O(n²), Space: O(n²) or O(n) with rolling array. Equivalent to LCS(s, reverse(s)).

07 — String to Integer / atoi (LC 8)

Handle: whitespace, sign, overflow, non-digit chars.

int myAtoi(string s) {
    int i = 0, n = s.size();
    int sign = 1;
    long result = 0;

    // 1. skip leading whitespace
    while (i < n && s[i] == ' ') i++;

    // 2. handle sign
    if (i < n && (s[i] == '+' || s[i] == '-')) {
        sign = (s[i] == '-') ? -1 : 1;
        i++;
    }

    // 3. convert digits, check overflow at each step
    while (i < n && s[i] >= '0' && s[i] <= '9') {
        result = result * 10 + (s[i] - '0');
        if (result * sign >= INT_MAX) return INT_MAX;
        if (result * sign <= INT_MIN) return INT_MIN;
        i++;
    }

    return (int)(result * sign);
}
  • Edge cases: empty string, only whitespace, sign with no digits, overflow mid-parse
  • Key: check overflow INSIDE the loop, not after

08 — Implement strStr / Find Index (LC 28)

KMP application — find first occurrence of needle in haystack.

int strStr(string haystack, string needle) {
    int n = haystack.size(), m = needle.size();
    if (m == 0) return 0;

    // build LPS for needle
    vector<int> lps(m, 0);
    int len = 0, i = 1;
    while (i < m) {
        if (needle[i] == needle[len]) {
            lps[i++] = ++len;
        } else if (len > 0) {
            len = lps[len - 1];
        } else {
            lps[i++] = 0;
        }
    }

    // KMP search
    i = 0;
    int j = 0;
    while (i < n) {
        if (haystack[i] == needle[j]) {
            i++; j++;
        }
        if (j == m) {
            return i - j;  // first occurrence
        } else if (i < n && haystack[i] != needle[j]) {
            if (j > 0) j = lps[j - 1];
            else i++;
        }
    }
    return -1;
}

09 — Repeated Substring Pattern (LC 459)

Check if string can be formed by repeating a substring.

bool repeatedSubstringPattern(string s) {
    int n = s.size();

    // Build LPS array
    vector<int> lps(n, 0);
    int len = 0, i = 1;
    while (i < n) {
        if (s[i] == s[len]) {
            lps[i++] = ++len;
        } else if (len > 0) {
            len = lps[len - 1];
        } else {
            lps[i++] = 0;
        }
    }

    // Key insight: if lps[n-1] > 0 and n is divisible by (n - lps[n-1]),
    // then the string is formed by repeating s[0..n-lps[n-1]-1]
    int patLen = n - lps[n - 1];
    return lps[n - 1] > 0 && n % patLen == 0;
}

// Alternative one-liner trick (less efficient but elegant):
bool repeatedSubstringPatternV2(string s) {
    string doubled = s + s;
    // remove first and last char, then search for s
    return doubled.substr(1, doubled.size() - 2).find(s) != string::npos;
}
KMP trick: patLen = n - lps[n-1] is the shortest repeating unit. If n % patLen == 0, the whole string is built from it.
