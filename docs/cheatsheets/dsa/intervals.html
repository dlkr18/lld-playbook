Intervals // Merge & Sweep

Sort by start or end, merge greedily, or decompose into events. These patterns cover every interval problem in interviews.

Sort + Merge Greedy Sweep Line Priority Queue

01 — Pattern Guide

PatternWhen to UseKey Idea
Merge Combine overlapping intervals Sort by start, extend end greedily
Insert Add interval to sorted list Find overlap window, merge in-place
Overlap / Greedy Max non-overlapping, min removals Sort by end, greedy keep earliest-ending
Sweep Line Max concurrent, count at point +1 at start, −1 at end, running sum
Rule of thumb: Sort by start for merge problems. Sort by end for greedy/removal problems. Use sweep line when you need the count at any point.

02 — Merge Intervals (LC 56)

Given an array of intervals, merge all overlapping intervals.

Approach

  • Sort by start time
  • Iterate: if current overlaps with last merged, extend end
  • Otherwise push new interval
#include <vector>
#include <algorithm>
using namespace std;

vector<vector<int>> merge(vector<vector<int>>& intervals) {
    sort(intervals.begin(), intervals.end());
    vector<vector<int>> res;
    for (auto& iv : intervals) {
        if (!res.empty() && res.back()[1] >= iv[0]) {
            res.back()[1] = max(res.back()[1], iv[1]);
        } else {
            res.push_back(iv);
        }
    }
    return res;
}

Time: O(n log n) • Space: O(n)

03 — Insert Interval (LC 57)

Insert a new interval into a sorted non-overlapping list, merging if necessary.

Approach

  • Add all intervals ending before newInterval starts
  • Merge all overlapping with newInterval
  • Add remaining intervals
vector<vector<int>> insert(vector<vector<int>>& intervals,
                             vector<int>& newInterval) {
    vector<vector<int>> res;
    int i = 0, n = intervals.size();

    // Add all intervals ending before new starts
    while (i < n && intervals[i][1] < newInterval[0])
        res.push_back(intervals[i++]);

    // Merge overlapping intervals
    while (i < n && intervals[i][0] <= newInterval[1]) {
        newInterval[0] = min(newInterval[0], intervals[i][0]);
        newInterval[1] = max(newInterval[1], intervals[i][1]);
        i++;
    }
    res.push_back(newInterval);

    // Add remaining
    while (i < n)
        res.push_back(intervals[i++]);

    return res;
}

Time: O(n) • Space: O(n)

04 — Non-overlapping Intervals (LC 435)

Find the minimum number of intervals to remove to make the rest non-overlapping.

Approach

  • Sort by end time (greedy: keep the one that ends earliest)
  • If current start < previous end → overlap → remove (increment count)
  • Otherwise update previous end
int eraseOverlapIntervals(vector<vector<int>>& intervals) {
    sort(intervals.begin(), intervals.end(),
         [](const vector<int>& a, const vector<int>& b) {
             return a[1] < b[1];
         });

    int removals = 0;
    int prevEnd = intervals[0][1];

    for (int i = 1; i < intervals.size(); i++) {
        if (intervals[i][0] < prevEnd) {
            removals++;  // overlap → remove current (it ends later)
        } else {
            prevEnd = intervals[i][1];
        }
    }
    return removals;
}
Why sort by end? Keeping the interval that ends earliest leaves the most room for future intervals — classic activity selection / greedy proof.

05 — Meeting Rooms (LC 252)

Can a person attend all meetings? (No overlaps)

bool canAttendMeetings(vector<vector<int>>& intervals) {
    sort(intervals.begin(), intervals.end());
    for (int i = 1; i < intervals.size(); i++) {
        if (intervals[i][0] < intervals[i - 1][1])
            return false;
    }
    return true;
}

Time: O(n log n) • Space: O(1)

06 — Meeting Rooms II (LC 253)

Find the minimum number of conference rooms required.

Solution A: Min-Heap

#include <queue>

int minMeetingRooms(vector<vector<int>>& intervals) {
    sort(intervals.begin(), intervals.end());
    priority_queue<int, vector<int>, greater<int>> pq; // min-heap of end times

    for (auto& iv : intervals) {
        if (!pq.empty() && pq.top() <= iv[0])
            pq.pop();  // reuse earliest-ending room
        pq.push(iv[1]);
    }
    return pq.size();
}

Solution B: Sweep Line

int minMeetingRooms(vector<vector<int>>& intervals) {
    vector<pair<int,int>> events;
    for (auto& iv : intervals) {
        events.push_back({iv[0], +1});  // meeting starts
        events.push_back({iv[1], -1});  // meeting ends
    }
    sort(events.begin(), events.end());

    int rooms = 0, maxRooms = 0;
    for (auto& [time, delta] : events) {
        rooms += delta;
        maxRooms = max(maxRooms, rooms);
    }
    return maxRooms;
}

Both: Time O(n log n) • Space O(n)

07 — Minimum Number of Arrows (LC 452)

Burst all balloons with minimum arrows. Each balloon is an interval [x_start, x_end].

Approach

  • Sort by end (same greedy as activity selection)
  • Shoot arrow at current end
  • Skip all balloons that overlap with this arrow
int findMinArrowShots(vector<vector<int>>& points) {
    sort(points.begin(), points.end(),
         [](const vector<int>& a, const vector<int>& b) {
             return a[1] < b[1];
         });

    int arrows = 1;
    int arrowPos = points[0][1];

    for (int i = 1; i < points.size(); i++) {
        if (points[i][0] > arrowPos) {
            arrows++;
            arrowPos = points[i][1];
        }
    }
    return arrows;
}
Key insight: This is the dual of LC 435. Min arrows = total intervals − max non-overlapping removals. Same greedy, different framing.

08 — Sweep Line Template

Generic event-based processing for interval problems that ask "how many active at any point?"

// Generic sweep line template
#include <vector>
#include <algorithm>
#include <map>
using namespace std;

int sweepLine(vector<vector<int>>& intervals) {
    map<int, int> events;  // sorted by time automatically

    for (auto& iv : intervals) {
        events[iv[0]]++;   // +1 at start
        events[iv[1]]--;   // -1 at end
    }

    int active = 0, maxActive = 0;
    for (auto& [time, delta] : events) {
        active += delta;
        maxActive = max(maxActive, active);
    }
    return maxActive;
}

// Variant: use vector of pairs when time range is bounded
int sweepLineVector(vector<vector<int>>& intervals) {
    vector<pair<int,int>> events;
    for (auto& iv : intervals) {
        events.push_back({iv[0], +1});
        events.push_back({iv[1], -1});
    }
    sort(events.begin(), events.end());

    int active = 0, maxActive = 0;
    for (auto& [time, delta] : events) {
        active += delta;
        maxActive = max(maxActive, active);
    }
    return maxActive;
}

When to use which variant

VariantUse WhenAdvantage
map Sparse time range, coordinate compression not wanted Auto-sorted, handles large coordinates
vector<pair> Need custom tie-breaking or multiple event types Faster sort, explicit control
